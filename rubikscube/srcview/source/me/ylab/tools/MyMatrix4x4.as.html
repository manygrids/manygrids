package me.ylab.tools {

    /*
        0 4  8 12
        1 5  9 13
        2 6 10 14
        3 7 11 15
    */
    public class MyMatrix4x4 {

        protected var _m:Vector.<Number> = new Vector.<Number>(16);

        public function MyMatrix4x4() {
            identify();
        }
        
        public function get m():Vector.<Number> {
            return _m;
        }
        
        public function equals(m:MyMatrix4x4):Boolean {
            for (var i:int=0; i<_m.length; i++) {
                if (_m[i] != m._m[i]) return false;
            }
            return true;
        }
        
        public function toString():String {
            var res:String = '';
            res += _m[0] + ' ' + _m[4] + ' ' + _m[8] + ' ' + _m[12] + '\n';
            res += _m[1] + ' ' + _m[5] + ' ' + _m[9] + ' ' + _m[13] + '\n';
            res += _m[2] + ' ' + _m[6] + ' ' + _m[10] + ' ' + _m[14] + '\n';
            res += _m[3] + ' ' + _m[7] + ' ' + _m[11] + ' ' + _m[15] + '\n';
            return res;
        }
        
        public function saveToString():String {
            var res:String = '[';
            res += _m[0] + ',' + _m[4] + ',' + _m[8] + ',' + _m[12] + ',';
            res += _m[1] + ',' + _m[5] + ',' + _m[9] + ',' + _m[13] + ',';
            res += _m[2] + ',' + _m[6] + ',' + _m[10] + ',' + _m[14] + ',';
            res += _m[3] + ',' + _m[7] + ',' + _m[11] + ',' + _m[15] + ']';
            return res;
        }
        
        public function loadFromString(str:String):void {
            
        }
        
        public function identify():void {
            for (var i:int=0; i<16; i++) {
                _m[i] = 0;
            }
            _m[0] = _m[5] = _m[10] = _m[15] = 1;
        }
        
        public function clearRotate():void {
            var tx:Number = _m[12];
            var ty:Number = _m[13];
            var tz:Number = _m[14];

            identify();
            _m[12] = tx;
            _m[13] = ty;
            _m[14] = tz;
        }
        
        public function addTranslate(x:Number, y:Number, z:Number):MyMatrix4x4 {
            var transMatrix:MyMatrix4x4 = MyMatrix4x4.getTranslateMatrix(x, y, z);
            transMatrix.multiply(this);
            for (var i:int=0; i<16; i++) {
                _m[i] = transMatrix.m[i];
            }
            return this;
        }
        
        public function addRotateByX(ang:Number):MyMatrix4x4 {
            var rotateMatrix:MyMatrix4x4 = MyMatrix4x4.getRotateByXMatrix(ang);
            rotateMatrix.multiply(this);
            for (var i:int=0; i<16; i++) {
                _m[i] = rotateMatrix.m[i];
            }
            return this;
        }
        
        public function addRotateByY(ang:Number):MyMatrix4x4 {
            var rotateMatrix:MyMatrix4x4 = MyMatrix4x4.getRotateByYMatrix(ang);
            rotateMatrix.multiply(this);
            for (var i:int=0; i<16; i++) {
                _m[i] = rotateMatrix.m[i];
            }
            return this;
        }
        
        public function addRotateByZ(ang:Number):MyMatrix4x4 {
            var rotateMatrix:MyMatrix4x4 = MyMatrix4x4.getRotateByZMatrix(ang);
            rotateMatrix.multiply(this);
            for (var i:int=0; i<16; i++) {
                _m[i] = rotateMatrix.m[i];
            }
            return this;
        }
        
        public function addScale(x:Number, y:Number, z:Number):MyMatrix4x4 {
            var scaleMatrix:MyMatrix4x4 = MyMatrix4x4.getScaleMatrix(x, y, z);
            scaleMatrix.multiply(this);
            for (var i:int=0; i<16; i++) {
                _m[i] = scaleMatrix.m[i];
            }
            return this;
        }
        
        public static function getTranslateMatrix(x:Number, y:Number, z:Number):MyMatrix4x4 {
            var res:MyMatrix4x4 = new MyMatrix4x4;
            res.m[12] = x;
            res.m[13] = y;
            res.m[14] = z;
            return res;
        }
        
        public static function getRotateByXMatrix(ang:Number):MyMatrix4x4 {
            var radian:Number = Math.PI * ang / 180;
            var res:MyMatrix4x4 = new MyMatrix4x4;
            res.m[5] = res.m[10] = Math.cos(radian);
            res.m[6] = res.m[9] = Math.sin(radian);
            res.m[9] = -res.m[9];
            return res;
        }
        
        public static function getRotateByYMatrix(ang:Number):MyMatrix4x4 {
            var radian:Number = Math.PI * ang / 180;
            var res:MyMatrix4x4 = new MyMatrix4x4;
            res.m[0] = res.m[10] = Math.cos(radian);
            res.m[2] = res.m[8] = Math.sin(radian);
            res.m[2] = -res.m[2];
            return res;
        }
        
        public static function getRotateByZMatrix(ang:Number):MyMatrix4x4 {
            var radian:Number = Math.PI * ang / 180;
            var res:MyMatrix4x4 = new MyMatrix4x4;
            res.m[0] = res.m[5] = Math.cos(radian);
            res.m[1] = res.m[4] = Math.sin(radian);
            res.m[4] = -res.m[4];
            return res;
        }
        
        public static function getScaleMatrix(x:Number, y:Number, z:Number):MyMatrix4x4 {
            var res:MyMatrix4x4 = new MyMatrix4x4;
            res.m[0] = x;
            res.m[5] = y;
            res.m[10] = z;
            return res;
        }
        
        public static function Multiply(m1:MyMatrix4x4, m2:MyMatrix4x4):MyMatrix4x4 {
            var res:MyMatrix4x4 = new MyMatrix4x4;
            
            res.m[0]  = m1.m[0] * m2.m[0] + m1.m[4] * m2.m[1] + m1.m[8] * m2.m[2]  + m1.m[12] * m2.m[3];
            res.m[1]  = m1.m[1] * m2.m[0] + m1.m[5] * m2.m[1] + m1.m[9] * m2.m[2]  + m1.m[13] * m2.m[3];
            res.m[2]  = m1.m[2] * m2.m[0] + m1.m[6] * m2.m[1] + m1.m[10] * m2.m[2] + m1.m[14] * m2.m[3];
            res.m[3]  = m1.m[3] * m2.m[0] + m1.m[7] * m2.m[1] + m1.m[11] * m2.m[2] + m1.m[15] * m2.m[3];

            res.m[4]  = m1.m[0] * m2.m[4] + m1.m[4] * m2.m[5] + m1.m[8] * m2.m[6]  + m1.m[12] * m2.m[7];
            res.m[5]  = m1.m[1] * m2.m[4] + m1.m[5] * m2.m[5] + m1.m[9] * m2.m[6]  + m1.m[13] * m2.m[7];
            res.m[6]  = m1.m[2] * m2.m[4] + m1.m[6] * m2.m[5] + m1.m[10] * m2.m[6] + m1.m[14] * m2.m[7];
            res.m[7]  = m1.m[3] * m2.m[4] + m1.m[7] * m2.m[5] + m1.m[11] * m2.m[6] + m1.m[15] * m2.m[7];

            res.m[8]  = m1.m[0] * m2.m[8] + m1.m[4] * m2.m[9] + m1.m[8] * m2.m[10]  + m1.m[12] * m2.m[11];
            res.m[9]  = m1.m[1] * m2.m[8] + m1.m[5] * m2.m[9] + m1.m[9] * m2.m[10]  + m1.m[13] * m2.m[11];
            res.m[10]  = m1.m[2] * m2.m[8] + m1.m[6] * m2.m[9] + m1.m[10] * m2.m[10] + m1.m[14] * m2.m[11];
            res.m[11]  = m1.m[3] * m2.m[8] + m1.m[7] * m2.m[9] + m1.m[11] * m2.m[10] + m1.m[15] * m2.m[11];

            res.m[12]  = m1.m[0] * m2.m[12] + m1.m[4] * m2.m[13] + m1.m[8] * m2.m[14]  + m1.m[12] * m2.m[15];
            res.m[13]  = m1.m[1] * m2.m[12] + m1.m[5] * m2.m[13] + m1.m[9] * m2.m[14]  + m1.m[13] * m2.m[15];
            res.m[14]  = m1.m[2] * m2.m[12] + m1.m[6] * m2.m[13] + m1.m[10] * m2.m[14] + m1.m[14] * m2.m[15];
            res.m[15]  = m1.m[3] * m2.m[12] + m1.m[7] * m2.m[13] + m1.m[11] * m2.m[14] + m1.m[15] * m2.m[15];
            
            return res;
        }
        
        public function multiply(matrix:MyMatrix4x4):MyMatrix4x4 {
            var res:MyMatrix4x4 = new MyMatrix4x4;
            
            res.m[0]  = _m[0] * matrix.m[0] + _m[4] * matrix.m[1] + _m[8] * matrix.m[2]  + _m[12] * matrix.m[3];
            res.m[1]  = _m[1] * matrix.m[0] + _m[5] * matrix.m[1] + _m[9] * matrix.m[2]  + _m[13] * matrix.m[3];
            res.m[2]  = _m[2] * matrix.m[0] + _m[6] * matrix.m[1] + _m[10] * matrix.m[2] + _m[14] * matrix.m[3];
            res.m[3]  = _m[3] * matrix.m[0] + _m[7] * matrix.m[1] + _m[11] * matrix.m[2] + _m[15] * matrix.m[3];
            
            res.m[4]  = _m[0] * matrix.m[4] + _m[4] * matrix.m[5] + _m[8] * matrix.m[6]  + _m[12] * matrix.m[7];
            res.m[5]  = _m[1] * matrix.m[4] + _m[5] * matrix.m[5] + _m[9] * matrix.m[6]  + _m[13] * matrix.m[7];
            res.m[6]  = _m[2] * matrix.m[4] + _m[6] * matrix.m[5] + _m[10] * matrix.m[6] + _m[14] * matrix.m[7];
            res.m[7]  = _m[3] * matrix.m[4] + _m[7] * matrix.m[5] + _m[11] * matrix.m[6] + _m[15] * matrix.m[7];
            
            res.m[8]  = _m[0] * matrix.m[8] + _m[4] * matrix.m[9] + _m[8] * matrix.m[10]  + _m[12] * matrix.m[11];
            res.m[9]  = _m[1] * matrix.m[8] + _m[5] * matrix.m[9] + _m[9] * matrix.m[10]  + _m[13] * matrix.m[11];
            res.m[10]  = _m[2] * matrix.m[8] + _m[6] * matrix.m[9] + _m[10] * matrix.m[10] + _m[14] * matrix.m[11];
            res.m[11]  = _m[3] * matrix.m[8] + _m[7] * matrix.m[9] + _m[11] * matrix.m[10] + _m[15] * matrix.m[11];
            
            res.m[12]  = _m[0] * matrix.m[12] + _m[4] * matrix.m[13] + _m[8] * matrix.m[14]  + _m[12] * matrix.m[15];
            res.m[13]  = _m[1] * matrix.m[12] + _m[5] * matrix.m[13] + _m[9] * matrix.m[14]  + _m[13] * matrix.m[15];
            res.m[14]  = _m[2] * matrix.m[12] + _m[6] * matrix.m[13] + _m[10] * matrix.m[14] + _m[14] * matrix.m[15];
            res.m[15]  = _m[3] * matrix.m[12] + _m[7] * matrix.m[13] + _m[11] * matrix.m[14] + _m[15] * matrix.m[15];
            
            for (var i:int=0; i<16; i++) {
                _m[i] = res.m[i];
            }
            
            return this;
        }

        public static function MultiplyVertex(matrix:MyMatrix4x4, v:MyVertex4):MyVertex4 {
            var res:MyVertex4 = new MyVertex4;
            res.setXYZW(v.x * matrix.m[0] + v.y * matrix.m[4] + v.z * matrix.m[8]  + v.w * matrix.m[12],
                        v.x * matrix.m[1] + v.y * matrix.m[5] + v.z * matrix.m[9]  + v.w * matrix.m[13],
                        v.x * matrix.m[2] + v.y * matrix.m[6] + v.z * matrix.m[10] + v.w * matrix.m[14],
                        v.x * matrix.m[3] + v.y * matrix.m[7] + v.z * matrix.m[11] + v.w * matrix.m[15]);
            return res;
        }
        
        public function multiplyVertex(v:MyVertex4):MyVertex4 {
            return MultiplyVertex(this, v);
        }
        
        public function multiplyToVertex(v:MyVertex4):void {
            var x:Number = v.x;
            var y:Number = v.y;
            var z:Number = v.z;
            var w:Number = v.w;
            v.setXYZW(x * _m[0] + y * _m[4] + z * _m[8]  + w * _m[12],
                      x * _m[1] + y * _m[5] + z * _m[9]  + w * _m[13],
                      x * _m[2] + y * _m[6] + z * _m[10] + w * _m[14],
                      x * _m[3] + y * _m[7] + z * _m[11] + w * _m[15]);
        }
        
        public function multiplyVertexAtoB(va:MyVertex4, vb:MyVertex4):void {
            vb.setXYZ(va.x * _m[0] + va.y * _m[4] + va.z * _m[8]  + _m[12],
                      va.x * _m[1] + va.y * _m[5] + va.z * _m[9]  + _m[13],
                      va.x * _m[2] + va.y * _m[6] + va.z * _m[10] + _m[14]);
        }
        
        public function resetBy(matrix:MyMatrix4x4):void {
            for (var i:int=0; i<16; i++) {
                _m[i] = matrix.m[i];
            }
        }

        public static function CreateFrom(matrix:MyMatrix4x4):MyMatrix4x4 {
            var res:MyMatrix4x4 = new MyMatrix4x4;
            for (var i:int=0; i<16; i++) {
                res.m[i] = matrix.m[i];
            }
            return res;
        }
        
        public static function GetRotateMatrix(matrix:MyMatrix4x4):MyMatrix4x4 {
            var res:MyMatrix4x4 = new MyMatrix4x4;
            for (var i:int=0; i<12; i++) {
                res.m[i] = matrix.m[i];
            }
            return res;
        }
    }
