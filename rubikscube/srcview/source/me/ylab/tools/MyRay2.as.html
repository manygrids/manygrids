package me.ylab.tools {

    public class MyRay2 extends MyLine2 {

        public function MyRay2() {
            super();
        }
        
        public function get start():MyVertex2 {
            return mCross;
        }

        public override function getVertexByParam(t:Number):MyVertex2 {
            if (t < 0) return null;
            var ret:MyVertex2 = new MyVertex2;
            ret.setXY( mCross.x + mDirection.x * t,
                       mCross.y + mDirection.y * t);
            return ret;
        }
        
        public override function fillVertexByParam(t:Number, v:MyVertex2):Boolean {
            if (t < 0) return false;
            v.setXY(mCross.x + mDirection.x * t,
                    mCross.y + mDirection.y * t);
            return true;
        }

    }
}
