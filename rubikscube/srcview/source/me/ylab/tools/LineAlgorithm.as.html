package me.ylab.tools {

    public class LineAlgorithm {

        public function LineAlgorithm() {
        }
        
        public static function bresenham(x1:int, y1:int, x2:int, y2:int, pointVector:Vector.<int>):void {
            var i:int = 0;
            var x:int = x1;
            var y:int = y1;
            var xStep:int = 1;
            var yStep:int = 1;
            var dx:int = Math.abs(x2 - x1);
            var dy:int = Math.abs(y2 - y1);
            if (x2 < x1) xStep = -1;
            if (y2 < y1) yStep = -1;
            if (x1 == x2) {
                for (i=0; i<dy; i++) {
                    pointVector.push(x);
                    pointVector.push(y);
                    y += yStep;
                }
                return;
            } else if (0 == dy) {
                for (i=0; i<dx; i++) {
                    pointVector.push(x);
                    pointVector.push(y);
                    x += xStep;
                }
                return;
            }
            
            var temp:int = 0;
            var swap:Boolean = false;
            if (dx < dy) {
                swap = true;
                temp = x1;
                x1 = y1;
                y1 = temp;
                temp = x2;
                x2 = y2;
                y2 = temp;
            }

            if (x2 < x1) {
                temp = x1;
                x1 = x2;
                x2 = temp;
                temp = y1;
                y1 = y2;
                y2 = temp;
            }
            
            x = x1;
            y = y1;
            
            dx = x2 - x1;
            dy = y2 - y1;
            
            yStep = 1;
            if (dy < 0) {
                yStep = -1;
                dy = -dy;
            }
            var dxdx:int = dx + dx;
            var dydy:int = dy + dy;
            var dydyMinusdxdx:int = dydy - dxdx;
            
            if (swap) {
                pointVector.push(y);
                pointVector.push(x);
            } else {
                pointVector.push(x);
                pointVector.push(y);
            }
            
            var p:int = dydy - dx;
            for (i=0; i<dx; i++) {
                if (p > 0) {
                    p -= dxdx;
                    y += yStep;
                }
                p += dydy;
                x++;
                if (swap) {
                    pointVector.push(y);
                    pointVector.push(x);
                } else {
                    pointVector.push(x);
                    pointVector.push(y);
                }
            }
        }
    }
}
