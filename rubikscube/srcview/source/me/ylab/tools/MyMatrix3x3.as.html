package me.ylab.tools {

    /*
        0 3 6
        1 4 7
        2 5 8
    */
    public class MyMatrix3x3 {

        private var _m:Vector.<Number> = new Vector.<Number>(9);
        
        public function MyMatrix3x3() {
            identify();
        }
        
        public function get m():Vector.<Number> {
            return _m;
        }

        public function identify():void {
            for (var i:Number=0; i<9; i++) {
                _m[i] = 0;
            }
            _m[0] = _m[4] = _m[8] = 1;
        }
        
        public function equals(m:MyMatrix3x3):Boolean {
            for (var i:int=0; i<_m.length; i++) {
                if (_m[i] != m._m[i]) return false;
            }
            return true;
        }
        
        public function addTranslate(x:Number, y:Number):MyMatrix3x3 {
            var transMatrix:MyMatrix3x3 = MyMatrix3x3.getTranslateMatrix(x, y);
            transMatrix.multiply(this);
            for (var i:int=0; i<9; i++) {
                _m[i] = transMatrix.m[i];
            }
            return this;
        }
        
        public function addRotate(ang:Number):MyMatrix3x3 {
            var rotateMatrix:MyMatrix3x3 = MyMatrix3x3.getRotateMatrix(ang);
            rotateMatrix.multiply(this);
            for (var i:int=0; i<9; i++) {
                _m[i] = rotateMatrix.m[i];
            }
            return this;
        }
        
        public function addScale(x:Number, y:Number):MyMatrix3x3 {
            var scaleMatrix:MyMatrix3x3 = MyMatrix3x3.getScaleMatrix(x, y);
            scaleMatrix.multiply(this);
            for (var i:int=0; i<9; i++) {
                _m[i] = scaleMatrix.m[i];
            }
            return this;
        }
        
        public static function getTranslateMatrix(x:Number, y:Number):MyMatrix3x3 {
            var res:MyMatrix3x3 = new MyMatrix3x3;
            res.m[6] = x;
            res.m[7] = y;
            return res;
        }
        
        public static function getRotateMatrix(ang:Number):MyMatrix3x3 {
            var radian:Number = Math.PI * ang / 180;
            var res:MyMatrix3x3 = new MyMatrix3x3;
            res.m[0] = res.m[4] = Math.cos(radian);
            res.m[1] = res.m[3] = Math.sin(radian);
            res.m[3] = -res.m[3];
            return res;
        }
        
        public static function getScaleMatrix(x:Number, y:Number):MyMatrix3x3 {
            var res:MyMatrix3x3 = new MyMatrix3x3;
            res.m[0] = x;
            res.m[4] = y;
            return res;
        }
        
        public static function Multiply(m1:MyMatrix3x3, m2:MyMatrix3x3):MyMatrix3x3 {
            var res:MyMatrix3x3 = new MyMatrix3x3;
            
            res.m[0] = m1.m[0] * m2.m[0] + m1.m[3] * m2.m[1] + m1.m[6] * m2.m[2];
            res.m[1] = m1.m[1] * m2.m[0] + m1.m[4] * m2.m[1] + m1.m[7] * m2.m[2];
            res.m[2] = m1.m[2] * m2.m[0] + m1.m[5] * m2.m[1] + m1.m[8] * m2.m[2];

            res.m[3] = m1.m[0] * m2.m[3] + m1.m[3] * m2.m[4] + m1.m[6] * m2.m[5];
            res.m[4] = m1.m[1] * m2.m[3] + m1.m[4] * m2.m[4] + m1.m[7] * m2.m[5];
            res.m[5] = m1.m[2] * m2.m[3] + m1.m[5] * m2.m[4] + m1.m[8] * m2.m[5];

            res.m[6] = m1.m[0] * m2.m[6] + m1.m[3] * m2.m[7] + m1.m[6] * m2.m[8];
            res.m[7] = m1.m[1] * m2.m[6] + m1.m[4] * m2.m[7] + m1.m[7] * m2.m[8];
            res.m[8] = m1.m[2] * m2.m[6] + m1.m[5] * m2.m[7] + m1.m[8] * m2.m[8];
            
            return res;
        }

        public function multiply(matrix:MyMatrix3x3):MyMatrix3x3 {
            var res:MyMatrix3x3 = new MyMatrix3x3;
            
            res.m[0] = _m[0] * matrix.m[0] + _m[3] * matrix.m[1] + _m[6] * matrix.m[2];
            res.m[1] = _m[1] * matrix.m[0] + _m[4] * matrix.m[1] + _m[7] * matrix.m[2];
            res.m[2] = _m[2] * matrix.m[0] + _m[5] * matrix.m[1] + _m[8] * matrix.m[2];
            
            res.m[3] = _m[0] * matrix.m[3] + _m[3] * matrix.m[4] + _m[6] * matrix.m[5];
            res.m[4] = _m[1] * matrix.m[3] + _m[4] * matrix.m[4] + _m[7] * matrix.m[5];
            res.m[5] = _m[2] * matrix.m[3] + _m[5] * matrix.m[4] + _m[8] * matrix.m[5];
            
            res.m[6] = _m[0] * matrix.m[6] + _m[3] * matrix.m[7] + _m[6] * matrix.m[8];
            res.m[7] = _m[1] * matrix.m[6] + _m[4] * matrix.m[7] + _m[7] * matrix.m[8];
            res.m[8] = _m[2] * matrix.m[6] + _m[5] * matrix.m[7] + _m[8] * matrix.m[8];

            for (var i:Number=0; i<9; i++) {
                _m[i] = res.m[i];
            }
            
            return this;
        }
        
        public static function MultiplyVector(matrix:MyMatrix3x3, v:MyVector3):MyVector3 {
            var res:MyVector3 = new MyVector3;
            res.setXYZ( v.x * matrix.m[0] + v.y * matrix.m[3] + v.z * matrix.m[6],
                        v.x * matrix.m[1] + v.y * matrix.m[4] + v.z * matrix.m[7],
                        v.x * matrix.m[2] + v.y * matrix.m[5] + v.z * matrix.m[8]);
            return res;
        }
        
        public function multiplyVector(v:MyVector3):MyVector3 {
            return MultiplyVector(this, v);
        }
        
        public static function CreateFrom(matrix:MyMatrix3x3):MyMatrix3x3 {
            var res:MyMatrix3x3 = new MyMatrix3x3;
            for (var i:int=0; i<9; i++) {
                res.m[i] = matrix.m[i];
            }
            return res;
        }

    }
}
