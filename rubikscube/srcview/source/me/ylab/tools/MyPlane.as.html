package me.ylab.tools {

    public class MyPlane {

        protected var mCross:MyVertex4;
        protected var mNormal:MyVector4;
        
        public function MyPlane() {
            mCross = new MyVertex4;
            mNormal = new MyVector4;
        }    
        
        public function get cross():MyVertex4 {
            return mCross;
        }
        
        public function get normal():MyVector4 {
            return mNormal;
        }

        public static function PlaneToVertexDistance(plane:MyPlane, v:MyVertex3):Number{
            return MyVector3.DotProduct(MyVertex3.Minus(plane.cross, v), plane.normal.equalVertex3());
        }
        
        public function distanceToVertex(v:MyVertex3):Number {
            return (mCross.x - v.x) * mNormal.x + (mCross.y - v.y) * mNormal.y + (mCross.z - v.z) * mNormal.z;
        }
        
        public function getCrossVertex(line:MyLine3):MyVertex4 {
            var dotProduct:Number = line.direction.dotProduct(mNormal);
            if (0 != dotProduct) {
                var t:Number = distanceToVertex(line.cross) / dotProduct;
                return line.getVertexByParam(t);
            }
            return null;
        }
    }
}
