package me.ylab.tools {

    public class MyTriangle extends MyPlane {

        protected var mPoint1:MyVertex4;
        protected var mPoint2:MyVertex4;
        
        private var tCross:MyVertex4;
        
        public function MyTriangle() {
            mPoint1 = new MyVertex4;
            mPoint2 = new MyVertex4;
            tCross = new MyVertex4;
        }
        
        public function refreshNormal():void {
            mNormal = MyVector4.CrossProduct(new MyVector4(mPoint1.x - mCross.x, mPoint1.y - mCross.y, mPoint1.z - mCross.z),
                                            new MyVector4(mPoint2.x - mPoint1.x, mPoint2.y - mPoint1.y, mPoint2.z - mPoint1.z));
        }
        
        public static function CreateFrom(tri:MyTriangle):MyTriangle {
            var ret:MyTriangle = new MyTriangle;
            ret.mCross.setXYZ(tri.mCross.x, tri.mCross.y, tri.mCross.z);
            ret.mNormal.setXYZ(tri.mNormal.x, tri.mNormal.y, tri.mNormal.z);
            ret.mPoint1.setXYZ(tri.mPoint1.x, tri.mPoint1.y, tri.mPoint1.z);
            ret.mPoint2.setXYZ(tri.mPoint2.x, tri.mPoint2.y, tri.mPoint2.z);
            return ret;
        }

        public function get p0():MyVertex3 {
            return mCross;
        }
        
        public function get p1():MyVertex3 {
            return mPoint1;
        }

        public function get p2():MyVertex3 {
            return mPoint2;
        }
        
        private function decideInnerVertexByVectorSum(v:MyVertex3):Boolean {
            
            var v1:MyVector3 = MyVertex3.Minus(v, cross);
            var v2:MyVector3 = MyVertex3.Minus(v, mPoint1);
            var v3:MyVector3 = MyVertex3.Minus(v, mPoint2);
            
            v1.normalize();
            v2.normalize();
            v3.normalize();
            
            var sum:MyVector3 = v1.add(v2).add(v3);
            
            return sum.module <= 1;
        }
        
        private function decideInnerVertexByVectorDir(v:MyVertex4):Boolean {
            
            var v1:MyVector4 = MyVertex4.Minus(mPoint1, mCross);
            var v2:MyVector4 = MyVertex4.Minus(v, mCross);
            
            if (MyVector4.DotProduct(MyVector4.CrossProduct(v1, v2), mNormal) < 0) return false;
            
            v1 = MyVertex4.Minus(mPoint2, mPoint1);
            v2 = MyVertex4.Minus(v, mPoint1);
            if (MyVector4.DotProduct(MyVector4.CrossProduct(v1, v2), mNormal) < 0) return false;
            
            v1 = MyVertex4.Minus(mCross, mPoint2);
            v2 = MyVertex4.Minus(v, mPoint2);
            if (MyVector4.DotProduct(MyVector4.CrossProduct(v1, v2), mNormal) < 0) return false;
            
            return true;
        }
        
        private function decideInnerVertexByUV(vertex:MyVertex3):Boolean {
            // refer to:  http://www.nowamagic.net/algorithm/algorithm_PointInTriangleTest.php
//            var v0:MyVector3 = MyVertex3.Minus(mPoint1, mCross);
//            var v1:MyVector3 = MyVertex3.Minus(mPoint2, mCross);
//            var v2:MyVector3 = MyVertex3.Minus(vertex, mCross);
//            
//            var dot00:Number = v0.dotProduct(v0);
//            var dot01:Number = v0.dotProduct(v1);
//            var dot02:Number = v0.dotProduct(v2);
//            var dot11:Number = v1.dotProduct(v1);
//            var dot12:Number = v1.dotProduct(v2);
            
            var v0x:Number = mPoint1.x - mCross.x;
            var v0y:Number = mPoint1.y - mCross.y;
            var v0z:Number = mPoint1.z - mCross.z;
            var v1x:Number = mPoint2.x - mCross.x;
            var v1y:Number = mPoint2.y - mCross.y;
            var v1z:Number = mPoint2.z - mCross.z;
            var v2x:Number = vertex.x - mCross.x;
            var v2y:Number = vertex.y - mCross.y;
            var v2z:Number = vertex.z - mCross.z;

            var dot00:Number = v0x * v0x + v0y * v0y + v0z * v0z;
            var dot01:Number = v0x * v1x + v0y * v1y + v0z * v1z;
            var dot02:Number = v0x * v2x + v0y * v2y + v0z * v2z;
            var dot11:Number = v1x * v1x + v1y * v1y + v1z * v1z;
            var dot12:Number = v1x * v2x + v1y * v2y + v1z * v2z;
            
            var d:Number = 1 / (dot00 * dot11 - dot01 * dot01);
            
            var u:Number = (dot11 * dot02 - dot01 * dot12) * d;
            if (u < 0 || u > 1) return false; // if u out of range, return directly
            
            var v:Number = (dot00 * dot12 - dot01 * dot02) * d;
            if (v < 0 || v > 1) return false; // if v out of range, return directly
            
            return u + v <= 1;
        }
        
        public function isInnerVertex(v:MyVertex3):Boolean {
            return decideInnerVertexByUV(v);
        }
        
        public override function getCrossVertex(line:MyLine3):MyVertex4 {
            var dotProduct:Number = line.direction.dotProduct(mNormal);
            if (dotProduct < 0) {
                var t:Number = distanceToVertex(line.cross) / dotProduct;
                if (line.fillVertexByParam(t, tCross) && isInnerVertex(tCross)) return tCross;
                return null;
            }
            return null;
        }

    }
}
