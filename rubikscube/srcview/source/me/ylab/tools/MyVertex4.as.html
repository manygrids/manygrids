package me.ylab.tools {

    public class MyVertex4 extends MyVertex3 {

        protected var mW:Number;

        public function MyVertex4(x:Number=0, y:Number=0, z:Number=0, w:Number = 1) {
            super(x, y, z);
            mW = w;
        }
        
        public function get w():Number {
            return mW;
        }
        
        
        public function set w(w:Number):void {
            mW = w;
        }
        
        public function equals(v:MyVertex4):Boolean {
            return (mX == v.x && mY == v.y && mZ == v.z && mW == v.w);
        }
        
        public function setXYZW(x:Number, y:Number, z:Number, w:Number):void {
            mX = x;
            mY = y;
            mZ = z;
            mW = w;
        }
        
        public function resetBy4(v:MyVertex4):void {
            mX = v.mX;
            mY = v.mY;
            mZ = v.mZ;
            mW = v.mW;
        }
        
        public static function Minus(v1:MyVertex4, v2:MyVertex4):MyVector4 {
            return new MyVector4(v1.mX - v2.mX, v1.mY - v2.mY, v1.mZ - v2.mZ);
        }
        
        public override function toString():String {
            return '[' + ((mX>-0.000001 && mX<0.000001)?0:mX) + ',' + ((mY>-0.000001 && mY<0.000001)?0:mY) + ',' + ((mZ>-0.000001 && mZ<0.000001)?0:mZ) + ',' + mW + ']';
        }
        
    }
}
