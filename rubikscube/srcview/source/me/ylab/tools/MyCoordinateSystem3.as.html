package me.ylab.tools {

    public class MyCoordinateSystem3 {
        
        protected var mSelfToWorld:MyMatrix4x4;
        protected var mWorldToSelf:MyMatrix4x4;

        public function MyCoordinateSystem3() {
            mSelfToWorld = new MyMatrix4x4;
            mWorldToSelf = new MyMatrix4x4;
        }
        
        public function insertTranslate(x:Number, y:Number, z:Number):void {
            mSelfToWorld.addTranslate(x, y, z);
            mWorldToSelf.multiply(MyMatrix4x4.getTranslateMatrix(-x, -y, -z));
        }
        
        public function insertRotateByX(ang:Number):void {
            mSelfToWorld.addRotateByX(ang);
            mWorldToSelf.multiply(MyMatrix4x4.getRotateByXMatrix(-ang));
        }
        
        public function insertRotateByY(ang:Number):void {
            mSelfToWorld.addRotateByY(ang);
            mWorldToSelf.multiply(MyMatrix4x4.getRotateByYMatrix(-ang));
        }
        
        public function insertRotateByZ(ang:Number):void {
            mSelfToWorld.addRotateByZ(ang);
            mWorldToSelf.multiply(MyMatrix4x4.getRotateByZMatrix(-ang));
        }
        
        public function addTranslate(x:Number, y:Number, z:Number):void {
            mSelfToWorld.multiply(MyMatrix4x4.getTranslateMatrix(x, y, z));
            mWorldToSelf.addTranslate(-x, -y, -z);
        }
        
        public function addRotateByX(ang:Number):void {
            mSelfToWorld.multiply(MyMatrix4x4.getRotateByXMatrix(ang));
            mWorldToSelf.addRotateByX(-ang);
        }
        
        public function addRotateByY(ang:Number):void {
            mSelfToWorld.multiply(MyMatrix4x4.getRotateByYMatrix(ang));
            mWorldToSelf.addRotateByY(-ang);
        }
        
        public function addRotateByZ(ang:Number):void {
            mSelfToWorld.multiply(MyMatrix4x4.getRotateByZMatrix(ang));
            mWorldToSelf.addRotateByZ(-ang);
        }
        
        public function get selfToWorld():MyMatrix4x4 {
            return mSelfToWorld;
        }

        public function get worldToSelf():MyMatrix4x4 {
            return mWorldToSelf;
        }
        
        public function returnToOrigin():void {
            mSelfToWorld.identify();
            mWorldToSelf.identify();
        }
        
        public function clearRotate():void {
            mSelfToWorld.clearRotate();
            mWorldToSelf.clearRotate();
        }
        
        public function exchangePos(coord:MyCoordinateSystem3):void {
            var tx:Number = mSelfToWorld.m[12];
            var ty:Number = mSelfToWorld.m[13];
            var tz:Number = mSelfToWorld.m[14];
            
            mSelfToWorld.m[12] = coord.mSelfToWorld.m[12];
            mSelfToWorld.m[13] = coord.mSelfToWorld.m[13];
            mSelfToWorld.m[14] = coord.mSelfToWorld.m[14];
            
            coord.mSelfToWorld.m[12] = tx;
            coord.mSelfToWorld.m[13] = ty;
            coord.mSelfToWorld.m[14] = tz;
            
            
            tx = mWorldToSelf.m[12];
            ty = mWorldToSelf.m[13];
            tz = mWorldToSelf.m[14];
            
            mWorldToSelf.m[12] = coord.mWorldToSelf.m[12];
            mWorldToSelf.m[13] = coord.mWorldToSelf.m[13];
            mWorldToSelf.m[14] = coord.mWorldToSelf.m[14];
            
            coord.mWorldToSelf.m[12] = tx;
            coord.mWorldToSelf.m[13] = ty;
            coord.mWorldToSelf.m[14] = tz;
            
        }
        
        public static function CreateFrom(coord:MyCoordinateSystem3):MyCoordinateSystem3 {
            var ret:MyCoordinateSystem3 = new MyCoordinateSystem3;
            ret.mSelfToWorld.resetBy(coord.mSelfToWorld);
            ret.mWorldToSelf.resetBy(coord.mWorldToSelf);
            return ret;
        }
        
    }
}
