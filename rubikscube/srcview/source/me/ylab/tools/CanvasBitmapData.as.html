package me.ylab.tools {
    import flash.display.BitmapData;

    public class CanvasBitmapData {

        private var mDrawCanvas:BitmapData;
        
        public function CanvasBitmapData(drawCanvas:BitmapData) {
            mDrawCanvas = drawCanvas;
        }
        
        public static function CanvasDrawLine(drawCanvas:BitmapData, x1:int, y1:int, x2:int, y2:int, color:int):void {
            var pv:Vector.<int> = new Vector.<int>;
            LineAlgorithm.bresenham(x1, y1, x2, y2, pv);
            for (var i:int=0; i<pv.length; ) {
                drawCanvas.setPixel(pv[i++], -pv[i++], color);
            }
        }

        public function canvasDrawLine(x1:int, y1:int, x2:int, y2:int, color:int):void {
            var pv:Vector.<int> = new Vector.<int>;
            LineAlgorithm.bresenham(x1, y1, x2, y2, pv);
            for (var i:int=0; i<pv.length; ) {
                mDrawCanvas.setPixel(pv[i++], -pv[i++], color);
            }
        }

    }
}
