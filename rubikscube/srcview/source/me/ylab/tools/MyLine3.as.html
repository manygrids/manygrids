package me.ylab.tools {

    public class MyLine3 {

        protected var mCross:MyVertex4;
        protected var mDirection:MyVector4;

        public function MyLine3() {
            mCross = new MyVertex4;
            mDirection = new MyVector4;
        }    

        public function get cross():MyVertex4 {
            return mCross;
        }
        
        public function get direction():MyVector4 {
            return mDirection;
        }
        
        public function getVertexByParam(t:Number):MyVertex4 {
            var ret:MyVertex4 = new MyVertex4;
            ret.setXYZ( mCross.x + mDirection.x * t,
                        mCross.y + mDirection.y * t,
                        mCross.z + mDirection.z * t);
            return ret;
        }
        
        public function fillVertexByParam(t:Number, v:MyVertex3):Boolean {
            v.setXYZ(mCross.x + mDirection.x * t,
                     mCross.y + mDirection.y * t,
                     mCross.z + mDirection.z * t);
            return true;
        }
    }
}
