package me.ylab.tools {

    public class MyVector3 extends MyVertex3 {

        public function MyVector3(x:Number=0, y:Number=0, z:Number=0) {
            super(x, y, z);
        }
        
        public function get module():Number {
            return Math.sqrt(mX * mX + mY * mY + mZ * mZ);
        }
        
        public function get moduleModule():Number {
            return mX * mX + mY * mY + mZ * mZ;
        }
        
        public function normalize():void {
            var vModule:Number = this.module;
            if (0 != vModule) {
                mX /= vModule;
                mY /= vModule;
                mZ /= vModule;
            }
        }
        
        public static function Add(v1:MyVector3, v2:MyVector3):MyVector3 {
            return new MyVector3(v1.x + v2.x, v1.y + v2.y, v1.z + v2.z);
        }
        
        public function add(v:MyVector3):MyVector3 {
            mX += v.x;
            mY += v.y;
            mZ += v.z;
            return this;
        }

        public static function Minus(v1:MyVector3, v2:MyVector3):MyVector3 {
            return new MyVector3(v1.x - v2.x, v1.y - v2.y, v1.z - v2.z);
        }
        
        public function minus(v:MyVector3):MyVector3 {
            mX -= v.x;
            mY -= v.y;
            mZ -= v.z;
            return this;
        }

        public function dotProduct(v:MyVector3):Number {
            return (mX * v.x + mY * v.y + mZ * v.z);
        }
        
        public static function DotProduct(v1:MyVector3, v2:MyVector3):Number {
            return (v1.x * v2.x + v1.y * v2.y + v1.z * v2.z);
        }
        
        public static function CrossProduct(v1:MyVector3, v2:MyVector3):MyVector3 {
            return new MyVector3(v1.y * v2.z - v2.y * v1.z, v1.z * v2.x - v2.z * v1.x, v1.x * v2.y - v2.x * v1.y);
        }
    }
}
