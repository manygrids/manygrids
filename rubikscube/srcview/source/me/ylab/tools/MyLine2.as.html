package me.ylab.tools {

    public class MyLine2 {
        
        public static const PRECISION:Number = 0.0000001; 

        protected var mCross:MyVertex2;
        protected var mDirection:MyVector2;

        public function MyLine2() {
            mCross = new MyVertex2;
            mDirection = new MyVector2;
        }    

        public function get cross():MyVertex2 {
            return mCross;
        }
        
        public function get direction():MyVector2 {
            return mDirection;
        }
        
        public function get range():MyVector2 {
            return mDirection;
        }
        
        public function distanceToVertex(v:MyVertex2):Number {
            return (v.x - mCross.x) * mDirection.y - (v.y - mCross.y) * mDirection.x;
        }
        
        public function getVertexByParam(t:Number):MyVertex2 {
            var ret:MyVertex2 = new MyVertex2;
            ret.setXY( mCross.x + mDirection.x * t,
                       mCross.y + mDirection.y * t);
            return ret;
        }
        
        public function fillVertexByParam(t:Number, v:MyVertex2):Boolean {
            v.setXY(mCross.x + mDirection.x * t,
                    mCross.y + mDirection.y * t);
            return true;
        }
        
        public function getCrossVertexWith(line:MyLine2):MyVertex2 {
            if (mDirection.equals2(line.direction)) {
                if (mCross.equals2(line.cross)) {
                    return new MyVertex2(mCross.x, mCross.y);
                } else {
                    var tCrossDir:Number = (mCross.y - line.cross.y) / (mCross.x - line.cross.x);
                    var tDir:Number = mDirection.y / mDirection.x;
                    if ((tCrossDir - tDir < PRECISION && tCrossDir - tDir > -PRECISION)
                        || (tCrossDir - tDir < -PRECISION && tCrossDir - tDir > PRECISION)) {
                        return new MyVertex2(mCross.x, mCross.y);
                    } else {
                        return null;
                    }
                }
            }
            var d:Number = line.direction.x*mDirection.y - mDirection.x*line.direction.y;
            if (d < PRECISION && d > -PRECISION) return null;
            
            var t1:Number = ((line.cross.x - mCross.x) * (-line.direction.y) - (-line.direction.x) * (line.cross.y - mCross.y)) / d;
            var t2:Number = (mDirection.x * (line.cross.y - mCross.y) - (line.cross.x - mCross.x) * mDirection.y) / d;
            var crossVertex:MyVertex2 = new MyVertex2;
            if (this.fillVertexByParam(t1, crossVertex)) {
                if (line.fillVertexByParam(t2, crossVertex)) return crossVertex;
            }
            return null;
        }
        
        public function toString():String {
            return mCross.toString() + ' -> ' + mDirection.toString();
        }
    }
}
