package me.ylab.application {
    import flash.utils.getTimer;

    public class MyAnimation {
        
        public static const ON_ANIMATION_START:int = 0;
        public static const ON_ANIMATION_END:int = 1;
        
        public var lastTime:int;
        public var nextTime:int;
        protected var mInterval:int;
        protected var mIsAnimated:Boolean;
        protected var mLoop:Boolean;
        protected var mEvents:Vector.<int>;
        protected var mEventListeners:Vector.<Vector.<Function>>;
        
        public function MyAnimation() {
            mIsAnimated = false;
            mEvents = new Vector.<int>;
            mEventListeners = new Vector.<Vector.<Function>>;
            mEvents.push(ON_ANIMATION_START);
            mEventListeners.push(new Vector.<Function>);
            mEvents.push(ON_ANIMATION_END);
            mEventListeners.push(new Vector.<Function>);
        }
        
        public function addEventListener(event:int, fun:Function):void {
            if (event < 0 || event >= mEvents.length) return;
            mEventListeners[event].push(fun);
        }
        
        public function removeEventListener(event:int, fun:Function):void {
            if (event < 0 || event >= mEvents.length) return;
            var pos:int = mEventListeners[event].indexOf(fun);
            if (pos < 0) return;
            mEventListeners[event].splice(pos, 1);
        }
        
        public function notifyEvent(event:int):void {
            if (event < 0 || event > mEvents.length) return;
            for each (var f:Function in mEventListeners[event]) {
                f.call(null, this);
            }
        }
        
        public function animate(lapseTime:int):void {}
        
        public function pause():void {}
        
        public function resume():void {}
        
        public function start():void {
            lastTime = getTimer();
            nextTime = lastTime + mInterval;
            mIsAnimated = true;
        }
        
        public function stop():void {
            nextTime = int.MAX_VALUE;
            mIsAnimated = false;
        }
        
        public function reset():void {
        }
        
        public function get interval():int {
            return mInterval;
        }
        
        public function set interval(i:int):void {
            mInterval = i;
        }
        
        public function get isAnimated():Boolean {
            return mIsAnimated;
        }
        
        public function get isLoop():Boolean {
            return mLoop;
        }
        
        public function set loop(l:Boolean):void {
            mLoop = l;
        }
    }
}
