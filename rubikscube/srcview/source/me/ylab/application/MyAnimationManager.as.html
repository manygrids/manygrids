package me.ylab.application {
    import flash.utils.getTimer;

    public class MyAnimationManager implements IRunable {

        protected var mAnimations:Vector.<MyAnimation>;
        
        public function MyAnimationManager() {
            mAnimations = new Vector.<MyAnimation>;
        }
        
        public function add(ani:MyAnimation):void {
            if (mAnimations.indexOf(ani) < 0) {
                mAnimations.push(ani);
            }
        }
        
        public function remove(ani:MyAnimation):void {
            var pos:int = mAnimations.indexOf(ani);
            if (pos < 0) return;
            mAnimations.splice(pos, 1);
        }
        
        public function run():void {
            var ct:int = 0;
            for each (var ani:MyAnimation in mAnimations) {
                ct = getTimer();
                if (ct >= ani.nextTime) {
                    ani.animate(ct - ani.lastTime);
                    ani.lastTime = ct;
                    while (ani.nextTime <= ct) ani.nextTime += ani.interval;
                }
            }
        }
        
    }
}
