package me.ylab.application {

    public class MyTimerAnimation extends MyAnimation {
        
        private var mLaspe:int;
        
        public function MyTimerAnimation(inter:int = 100) {
            super();
            mInterval = inter;
        }
        
        public override function animate(lapseTime:int):void {
            mLaspe = lapseTime;
            notifyEvent(ON_ANIMATION_END);
        }
        
        public function get laspe():int {
            return mLaspe;
        }
    }
}
