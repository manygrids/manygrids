package me.ylab.application
{
    import me.ylab.graphic.foundation.Graphic3D;

    internal class MyAppCommonRunner extends MyAppPauseRunner {
        
        protected var mLogic:MyLogic;
        
        public function MyAppCommonRunner(app:MyApplication, g:Graphic3D, l:MyLogic) {
            super(app, g);
            mLogic = l;
        }
        
        public override function run():void {
            mLogic.run(mApp.lapseTime);
            mGraphic.onRender();
        }
    }

}
