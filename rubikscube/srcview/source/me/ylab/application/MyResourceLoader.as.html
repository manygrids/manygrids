package me.ylab.application {

    public class MyResourceLoader {

        public static const LOAD_COMPLETE:int = 0;
        protected var mEventListeners:Vector.<Vector.<Function>>;
        protected var mResources:Vector.<IResource>;
        protected var mLoadedResCount:int;
        
        public function MyResourceLoader() {
            mEventListeners = new Vector.<Vector.<Function>>;
            mEventListeners[LOAD_COMPLETE] = new Vector.<Function>;
            mResources = new Vector.<IResource>;
        }
        
        public function addResource(res:IResource, delayLoad:Boolean = true):void {
            if (mResources.indexOf(res) < 0) {
                mResources.push(res);
                if (!delayLoad) {
                    if (!res.isLoaded()) {
                        res.addEventListener(MyResource.LOAD_COMPLETE, onLoadComplete);
                        res.load();
                    }
                }
            }
        }
        
        public function removeResource(res:IResource):void {
            var pos:int = mResources.indexOf(res);
            if (pos < 0) return;
            mResources.splice(pos, 1);
        }
        
        public function loadAll():void {
            mLoadedResCount = 0;
            for each (var res:IResource in mResources) {
                if (!res.isLoaded()) {
                    res.addEventListener(MyResource.LOAD_COMPLETE, onLoadComplete);
                    res.load();
                }
            }
        }
        
        protected function onLoadComplete(res:IResource):void {
            res.removeEventListener(MyResource.LOAD_COMPLETE, onLoadComplete);
            mLoadedResCount++;
            if (mLoadedResCount == mResources.length) {
                for each (var f:Function in mEventListeners[LOAD_COMPLETE]) {
                    f.call(null, this);
                }
            }
        }
        
        public function addEventListener(event:int, fun:Function):void {
            if (event < LOAD_COMPLETE || event > LOAD_COMPLETE) return;
            mEventListeners[event].push(fun);
        }
        
        public function removeEventListener(event:int, fun:Function):void {
            if (event < LOAD_COMPLETE || event > LOAD_COMPLETE) return;
            var pos:int = mEventListeners[event].indexOf(fun);
            if (pos < 0) return;
            mEventListeners[event].splice(pos, 1);
        }
    }
}
