package me.ylab.application {
    import me.ylab.graphic.foundation.Graphic3D;

    public class MyApplication {

        protected var mGraphic:Graphic3D;
        protected var mLogic:MyLogic;
        protected var mRunner:IRunable;
        
        protected var mEmptyRun:MyAppEmptyRunner;
        protected var mPauseRun:MyAppPauseRunner;
        protected var mCommonRun:MyAppCommonRunner;
        
        public function MyApplication(g:Graphic3D = null, l:MyLogic = null) {
            mEmptyRun = new MyAppEmptyRunner;
            mRunner = mEmptyRun;
            mLogic = l;
            mGraphic = g;
            if (null != mGraphic) {
                mPauseRun = new MyAppPauseRunner(this, mGraphic);
                mRunner = mPauseRun;
                if (null != mLogic) {
                    mCommonRun = new MyAppCommonRunner(this, mGraphic, mLogic);
                    mRunner = mCommonRun;
                }
            }
        }
        
        public function run():void {
            mRunner.run();
        }
        
        public function pause():void {
            if (null != mPauseRun)
                mRunner = mPauseRun;
            else
                mRunner = mEmptyRun;
        }
        
        public function resume():void {
            if (null != mCommonRun)
                mRunner = mCommonRun;
            else
                mRunner = mEmptyRun;
        }
        
        public function get graphic():Graphic3D {
            return mGraphic;
        }
        
        public function set graphic(g:Graphic3D):void {
            mGraphic = g;
            if (null != mLogic && null != mGraphic) {
                mCommonRun = new MyAppCommonRunner(this, mGraphic, mLogic);
            }
        }
        
        public function set logic(l:MyLogic):void {
            mLogic = l;
            if (null != mLogic) {
                if (mRunner == mCommonRun) {
                    mCommonRun = new MyAppCommonRunner(this, mGraphic, mLogic);
                    mRunner = mCommonRun;
                } else {
                    mCommonRun = new MyAppCommonRunner(this, mGraphic, mLogic);
                }
            } else {
                mRunner = mPauseRun;
            }
        }
        
        public function get logic():MyLogic {
            return mLogic;
        }
        
        public function get lapseTime():int {
            return 0;
        }
        
    }
}
