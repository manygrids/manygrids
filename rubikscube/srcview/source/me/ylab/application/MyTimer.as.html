package me.ylab.application {

    public class MyTimer {
        
        public static const ON_TIMER:int = 0;
        public static const ON_START:int = 1;
        public static const ON_PAUSE:int = 2;
        public static const ON_RESET:int = 3;
        
        protected var mEventListeners:Vector.<Vector.<Function>>;
        private var mLogic:MyLogic;
        private var mHour:int;
        private var mMinute:int;
        private var mSecond:int;
        private var mMillisecond:int;

        private var mDelegate:MyTimerAnimation;
        
        public function MyTimer(logic:MyLogic) {
            mEventListeners = new Vector.<Vector.<Function>>;
            mEventListeners[ON_TIMER] = new Vector.<Function>;
            mEventListeners[ON_START] = new Vector.<Function>;
            mEventListeners[ON_PAUSE] = new Vector.<Function>;
            mEventListeners[ON_RESET] = new Vector.<Function>;
            mLogic = logic;
            mDelegate = new MyTimerAnimation(100);
            mDelegate.addEventListener(MyAnimation.ON_ANIMATION_END, onTimer);
            mDelegate.stop();
            if (null != mLogic)
                mLogic.animationManager.add(mDelegate);
        }
        
        public function get hour():int {
            return mHour;
        }
        
        public function get second():int {
            return mSecond;
        }
        
        public function get minute():int {
            return mMinute;
        }
        
        public function reset():void {
            mHour = 0;
            mMinute = 0;
            mSecond = 0;
            notifyEvent(ON_RESET);
        }
        
        private function onTimer(ani:MyTimerAnimation):void {
            mMillisecond += ani.laspe;
            if (mMillisecond >= 1000) {
                mMillisecond -= 1000;
                mSecond++;
                if (60 == mSecond){
                    mSecond = 0;
                    mMinute++;
                    if (60 == mMinute) {
                        mHour++;
                    }
                }
                notifyEvent(ON_TIMER);
            }
        }
        
        public function start():void {
            mDelegate.start();
        }
        
        public function pause():void {
            mDelegate.stop();
        }
        
        public function addEventListener(event:int, fun:Function):void {
            if (event < ON_TIMER || event > ON_RESET) return;
            if (mEventListeners[event].indexOf(fun) < 0) {
                mEventListeners[event].push(fun);
            }
        }

        public function removeEventListener(event:int, fun:Function):void {
            if (event < ON_TIMER || event > ON_RESET) return;
            var pos:int = mEventListeners[event].indexOf(fun);
            if (pos < 0) return;
            mEventListeners[event].splice(pos, 1);
        }
        
        public function notifyEvent(event:int):void {
            if (event < ON_TIMER || event > ON_RESET) return;
            for each (var f:Function in mEventListeners[event]) {
                f.call(null, this);
            }
        }
    }
}
