package me.ylab.graphic.component {
    
    import flash.display3D.Context3D;
    import flash.display3D.Context3DVertexBufferFormat;
    
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.foundation.MyColor4f;
    import me.ylab.graphic.foundation.MyCubeTexture;
    import me.ylab.graphic.foundation.MyIndexBuffer;
    import me.ylab.graphic.foundation.MyModel;
    import me.ylab.graphic.foundation.MyTexture;
    import me.ylab.graphic.foundation.MyVertexBuffer;
    import me.ylab.graphic.types.GraphicVertex3;
    import me.ylab.graphic.types.indices.MyVertexIndicesMesh;
    import me.ylab.graphic.types.indices.cubetexture.MyVertexIndicesWithCubeTextureMesh;
    import me.ylab.graphic.types.indices.cubetexture.MyVertexIndicesWithCubeTextureTriangle;
    import me.ylab.tools.MyVertex3;
    
    public class CompCube extends MyModel {
        
        protected var mTexture:MyCubeTexture;
        
        public function CompCube(graphic:Graphic3D=null,
                                 lbb:MyVertex3=null,
                                 rtf:MyVertex3=null,
                                 color:MyColor4f=null,
                                 faceColor:Vector.<MyColor4f> = null,
                                 vertexBuffer:MyVertexBuffer=null,
                                 indexBuffer:MyIndexBuffer=null,
                                 texture:MyCubeTexture = null,
                                 autoInit:Boolean=true) {
            super(graphic);

            mTexture = texture;
            
            if (null == lbb) lbb = new MyVertex3;
            if (null == rtf) rtf = new MyVertex3(1, 1, 1);
            
            if (!autoInit) return;

            var mesh0:MyVertexIndicesMesh = new MyVertexIndicesMesh(graphic, color, vertexBuffer, indexBuffer);

            if (0 == mesh0.vertexBuffer.count) {
                mesh0.vertexBuffer.addGraphicVertex(new GraphicVertex3(lbb.x, lbb.y,  lbb.z));
                mesh0.vertexBuffer.addGraphicVertex(new GraphicVertex3(lbb.x, lbb.y,  rtf.z));
                mesh0.vertexBuffer.addGraphicVertex(new GraphicVertex3(lbb.x, rtf.y,  lbb.z));
                mesh0.vertexBuffer.addGraphicVertex(new GraphicVertex3(lbb.x, rtf.y,  rtf.z));
                
            }
            
            if (null == indexBuffer || 0 == indexBuffer.count) {
                mesh0.addTriangle(new MyVertexIndicesWithCubeTextureTriangle(0, 1, 2));
                mesh0.addTriangle(new MyVertexIndicesWithCubeTextureTriangle(2, 1, 3));
            }
            mesh0.createDrawBuffers();
            
            var mesh1:MyVertexIndicesMesh = MyVertexIndicesMesh.CreateFrom(mesh0);
            var mesh2:MyVertexIndicesMesh = MyVertexIndicesMesh.CreateFrom(mesh0);
            
            mesh1.coordinate.insertRotateByY(90);
            mesh1.coordinate.insertRotateByX(90);
            
            mesh2.coordinate.insertRotateByY(-90);
            mesh2.coordinate.insertRotateByZ(-90);
            
            var mesh3:MyVertexIndicesMesh = MyVertexIndicesMesh.CreateFrom(mesh0);
            var mesh4:MyVertexIndicesMesh = MyVertexIndicesMesh.CreateFrom(mesh1);
            var mesh5:MyVertexIndicesMesh = MyVertexIndicesMesh.CreateFrom(mesh2);
            
            mesh3.coordinate.insertRotateByY(180);
            mesh3.coordinate.insertRotateByX(-90);
            mesh3.coordinate.insertTranslate(rtf.x - lbb.x, rtf.y - lbb.y, rtf.z - lbb.z);
            
            mesh4.coordinate.insertRotateByY(180);
            mesh4.coordinate.insertRotateByX(-90);
            mesh4.coordinate.insertTranslate(rtf.x - lbb.x, rtf.y - lbb.y, rtf.z - lbb.z);
            
            mesh5.coordinate.insertRotateByY(180);
            mesh5.coordinate.insertRotateByX(-90);
            mesh5.coordinate.insertTranslate(rtf.x - lbb.x, rtf.y - lbb.y, rtf.z - lbb.z);

            addMesh(mesh0);
            addMesh(mesh1);
            addMesh(mesh2);
            addMesh(mesh3);
            addMesh(mesh4);
            addMesh(mesh5);

            if (null != faceColor) {
                for (var i:int=0; i<faceColor.length; i++) {
                    mMeshes[i].setColor(faceColor[i].r, faceColor[i].g, faceColor[i].b, faceColor[i].a);
                }
            }
            
        }
        
        public function bindVertexBuffer():void {
            MyVertexIndicesMesh(mMeshes[0]).bindVertexBuffer();
        }
        
        public function setFaceColor(index:int, color:MyColor4f):void {
            if (index < 0 || index > 5 || null == color) return;
            mMeshes[index].setColor(color.r, color.g, color.b, color.a);
        }
        
        public function getFaceColor(index:int):MyColor4f {
            return mMeshes[index].color; 
        }
        
        public static function CreateFrom(cube:CompCube):CompCube {
            var ret:CompCube = new CompCube(cube.mGraphic, null, null, null, null, null, null, cube.mTexture, false);
            for (var i:int=0; i<cube.mMeshes.length; i++) {
                ret.addMesh(MyVertexIndicesMesh.CreateFrom(MyVertexIndicesMesh(cube.mMeshes[i])));
            }
            
            ret.mPrepared = false;
            ret.mFather = cube.mFather;
            ret.mModels.concat(cube.mModels);
            ret.mCoordinate.selfToWorld.resetBy(cube.mCoordinate.selfToWorld);
            ret.mCoordinate.worldToSelf.resetBy(cube.mCoordinate.worldToSelf);
            
            return ret;
        }
    }
}
