package me.ylab.graphic.component {

    import me.ylab.graphic.types.GraphicVertex3;
    import me.ylab.graphic.types.indices.MyVertexIndicesMesh;
    import me.ylab.graphic.types.indices.MyVertexIndicesTriangle;
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.foundation.MyColor4f;
    import me.ylab.graphic.foundation.MyIndexBuffer;
    import me.ylab.graphic.foundation.MyModel;
    import me.ylab.graphic.foundation.MyVertexBuffer;
    import me.ylab.tools.MyVertex3;
    
    public class CompTriangle extends MyModel {

        public function CompTriangle(graphic:Graphic3D=null,
                                     p0:MyVertex3=null,
                                     p1:MyVertex3=null,
                                     p2:MyVertex3=null,
                                     color:MyColor4f=null,
                                     vertexBuffer:MyVertexBuffer=null,
                                     indexBuffer:MyIndexBuffer=null) {
            super(graphic);
            
            if (null == p0) p0 = new MyVertex3;
            if (null == p1) p1 = new MyVertex3(1, 0);
            if (null == p2) p2 = new MyVertex3(0, 1);
            
            var mesh:MyVertexIndicesMesh = new MyVertexIndicesMesh(graphic, color, vertexBuffer, indexBuffer);
            if (0 == mesh.vertexBuffer.count) {
                mesh.vertexBuffer.addGraphicVertex(new GraphicVertex3(p0.x, p0.y,  p0.z));
                mesh.vertexBuffer.addGraphicVertex(new GraphicVertex3(p1.x, p1.y,  p1.z));
                mesh.vertexBuffer.addGraphicVertex(new GraphicVertex3(p2.x, p2.y,  p2.z));
            }
            
            if (null == indexBuffer || 0 == indexBuffer.count) {
                mesh.addTriangle(new MyVertexIndicesTriangle(0, 1, 2));
            }
            
            addMesh(mesh);
        }
    }
}
