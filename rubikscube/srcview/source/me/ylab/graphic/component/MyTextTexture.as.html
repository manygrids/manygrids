package me.ylab.graphic.component {

    import flash.display.BitmapData;
    import flash.display3D.Context3DTextureFormat;
    import flash.text.TextField;
    import flash.text.TextFieldAutoSize;
    import flash.text.TextFormat;
    
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.foundation.MyColor4f;
    import me.ylab.graphic.foundation.MyTexture;
    
    public class MyTextTexture extends MyTexture {

        protected var mText:String;
        
        public function MyTextTexture(graphic:Graphic3D) {
            super(graphic);
            mText = new String;
        }
        
        public function get text():String {
            return mText;
        }
        
        public function setText(t:String, color:MyColor4f, size:int = 12, font:String = null):void {
            mText = t;
            var textFormat:TextFormat = new TextFormat();
            textFormat.color = color.rgb;
            textFormat.size = size;
            textFormat.font = font;
            
            var textField:TextField = new TextField();
            textField.defaultTextFormat = textFormat;
            textField.autoSize = TextFieldAutoSize.LEFT;
            textField.text = t;
            textField.x = 0;
            textField.y = 0;
            
            mWidth = 1;
            mHeight = 1;
            
            while (mWidth < textField.width) mWidth *= 2;
            while (mHeight < textField.height) mHeight *= 2;

            mWidthRatio = textField.width / mWidth;
            mHeightRatio = textField.height / mHeight;

            var transparentBD:BitmapData = new BitmapData(mWidth, mHeight, true, 0);
            transparentBD.draw(textField);
            
            mTexture = mGraphic.context.createTexture(mWidth, mHeight, Context3DTextureFormat.BGRA, false);
            
            mTexture.uploadFromBitmapData(transparentBD);
            transparentBD.dispose();
        }
    }
}
