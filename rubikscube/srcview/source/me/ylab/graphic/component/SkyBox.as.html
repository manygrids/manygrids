package me.ylab.graphic.component {
    
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.foundation.MyColor4f;
    import me.ylab.graphic.foundation.MyIndexBuffer;
    import me.ylab.graphic.foundation.MyModel;
    import me.ylab.graphic.foundation.MyTexture;
    import me.ylab.graphic.foundation.MyVertexBuffer;
    import me.ylab.graphic.types.GraphicVertex3;
    import me.ylab.graphic.types.TexturedVertex3;
    import me.ylab.graphic.types.indices.MyVertexIndicesTriangle;
    import me.ylab.graphic.types.indices.textured.MyTexturedVertexIndicesMesh;
    import me.ylab.graphic.types.indices.textured.MyTexturedVertexIndicesTriangle;
    import me.ylab.tools.MyVertex3;
    
    public class SkyBox extends MyModel {
        
        protected var mLbb:MyVertex3;
        protected var mRtf:MyVertex3;
        
        public function SkyBox(graphic:Graphic3D=null,
                                 lbb:MyVertex3=null,
                                 rtf:MyVertex3=null,
                                 color:MyColor4f=null,
                                 faceTexture:Vector.<MyTexture> = null,
                                 vertexBuffer:MyVertexBuffer=null,
                                 indexBuffer:MyIndexBuffer=null,
                                 autoInit:Boolean=true) {
            super(graphic);
            
            if (null == lbb) lbb = new MyVertex3;
            if (null == rtf) rtf = new MyVertex3(5, 5, 5);
            
            mLbb = lbb;
            mRtf = rtf;
            
            if (!autoInit) return;
            
            var mesh0:MyTexturedVertexIndicesMesh = new MyTexturedVertexIndicesMesh(graphic, vertexBuffer, indexBuffer, null);
            mesh0.transparent = false;
            
            if (0 == mesh0.vertexBuffer.count) {
                mesh0.vertexBuffer.addGraphicVertex(new TexturedVertex3(lbb.x, lbb.y,  lbb.z, 0, faceTexture[0].heightRatio));
                mesh0.vertexBuffer.addGraphicVertex(new TexturedVertex3(lbb.x, lbb.y,  rtf.z, faceTexture[0].widthRatio, faceTexture[0].heightRatio));
                mesh0.vertexBuffer.addGraphicVertex(new TexturedVertex3(lbb.x, rtf.y,  lbb.z, 0, 0));
                mesh0.vertexBuffer.addGraphicVertex(new TexturedVertex3(lbb.x, rtf.y,  rtf.z, faceTexture[0].widthRatio, 0));
            }
            
            if (null == indexBuffer || 0 == indexBuffer.count) {
                mesh0.addTriangle(new MyTexturedVertexIndicesTriangle(1, 0, 3));
                mesh0.addTriangle(new MyTexturedVertexIndicesTriangle(3, 0, 2));
            }
            
            var mesh1:MyTexturedVertexIndicesMesh = MyTexturedVertexIndicesMesh.CreateFrom(mesh0);
            var mesh2:MyTexturedVertexIndicesMesh = MyTexturedVertexIndicesMesh.CreateFrom(mesh0);
            
            mesh1.coordinate.insertRotateByY(90);
            mesh1.coordinate.insertRotateByX(90);
            
            mesh2.coordinate.insertRotateByY(-90);
            mesh2.coordinate.insertRotateByZ(-90);
            
            var mesh3:MyTexturedVertexIndicesMesh = MyTexturedVertexIndicesMesh.CreateFrom(mesh0);
            var mesh4:MyTexturedVertexIndicesMesh = MyTexturedVertexIndicesMesh.CreateFrom(mesh1);
            var mesh5:MyTexturedVertexIndicesMesh = MyTexturedVertexIndicesMesh.CreateFrom(mesh2);
            
            mesh3.coordinate.insertRotateByY(180);
            mesh3.coordinate.insertRotateByX(-90);
            mesh3.coordinate.insertTranslate(rtf.x - lbb.x, rtf.y - lbb.y, rtf.z - lbb.z);
            
            mesh4.coordinate.insertRotateByY(180);
            mesh4.coordinate.insertRotateByX(-90);
            mesh4.coordinate.insertTranslate(rtf.x - lbb.x, rtf.y - lbb.y, rtf.z - lbb.z);
            
            mesh5.coordinate.insertRotateByY(180);
            mesh5.coordinate.insertRotateByX(-90);
            mesh5.coordinate.insertTranslate(rtf.x - lbb.x, rtf.y - lbb.y, rtf.z - lbb.z);
            
            addMesh(mesh0);
            addMesh(mesh1);
            addMesh(mesh2);
            addMesh(mesh3);
            addMesh(mesh4);
            addMesh(mesh5);
            if (null != faceTexture) {
                for (var i:int=0; i<faceTexture.length; i++) {
                    MyTexturedVertexIndicesMesh(mMeshes[i]).texture = faceTexture[i];
                }
            }
            
            mCoordinate.insertTranslate(-(rtf.x - lbb.x) / 2, -(rtf.y - lbb.y) / 2, -(rtf.z - lbb.z) / 2);
        }

        public function resetTranslate():void {
            mCoordinate.addTranslate(-(mRtf.x - mLbb.x) / 2, -(mRtf.y - mLbb.y) / 2, -(mRtf.z - mLbb.z) / 2);
        }
        
        public static function CreateFrom(sb:SkyBox):SkyBox {
            var ret:SkyBox = new SkyBox(sb.mGraphic, null, null, null, null, null, null, false);
            for (var i:int=0; i<sb.mMeshes.length; i++) {
                ret.addMesh(MyTexturedVertexIndicesMesh.CreateFrom(MyTexturedVertexIndicesMesh(sb.mMeshes[i])));
            }
            
            ret.mPrepared = false;
            ret.mFather = sb.mFather;
            ret.mModels.concat(sb.mModels);
            ret.mCoordinate.selfToWorld.resetBy(sb.mCoordinate.selfToWorld);
            ret.mCoordinate.worldToSelf.resetBy(sb.mCoordinate.worldToSelf);
            
            return ret;
        }
    }
}
