package me.ylab.graphic.types {

    import flash.utils.ByteArray;
    import flash.utils.Endian;
    
    import me.ylab.tools.MyMatrix4x4;
    
    public class GraphicMatrix4x4 extends MyMatrix4x4 {

        public function GraphicMatrix4x4() {
            super();
        }

        public static function FromMatrix4x4(matrix:MyMatrix4x4):GraphicMatrix4x4 {
            var res:GraphicMatrix4x4 = new GraphicMatrix4x4();
            for (var i:int=0; i<16; i++) {
                res.m[i] = matrix.m[i];
            }
            return res;
        }
        
        public function copyFrom(matrix:MyMatrix4x4):void {
            for (var i:int=0; i<16; i++) {
                _m[i] = matrix.m[i];
            }
        }
        
        public function fillTransposeByteArray(ba:ByteArray):void {
            ba.clear();
            for (var j:int=0; j<4; j++) {
                for (var i:int=0; i<4; i++) {
                    ba.writeFloat(_m[j + i*4]);
                }
            }
        }
        
        public static function FillTransposeByteArray(ba:ByteArray, matrix:MyMatrix4x4):void {
            ba.clear();
            for (var j:int=0; j<4; j++) {
                for (var i:int=0; i<4; i++) {
                    ba.writeFloat(matrix.m[j + i*4]);
                }
            }
        }
        
        public static function FillByteArray(ba:ByteArray, matrix:MyMatrix4x4):void {
            ba.clear();
            for (var i:int=0; i<4; i++) {
                for (var j:int=0; j<4; j++) {
                    ba.writeFloat(matrix.m[j + i*4]);
                }
            }
        }
        
        public function get byteArray():ByteArray {
            var ba:ByteArray = new ByteArray();
            ba.endian = Endian.LITTLE_ENDIAN;
            for (var i:int=0; i<16; i++) {
                ba.writeFloat(_m[i]);
            }
            return ba;
        }

        public function get transposeByteArray():ByteArray {
            var ba:ByteArray = new ByteArray();
            ba.endian = Endian.LITTLE_ENDIAN;
            for (var j:int=0; j<4; j++) {
                for (var i:int=0; i<4; i++) {
                    ba.writeFloat(_m[j + i*4]);
                }
            }
            return ba;
        }
        
        public static function ToTransposeByteArray(matrix:MyMatrix4x4):ByteArray {
            var ba:ByteArray = new ByteArray();
            ba.endian = Endian.LITTLE_ENDIAN;
            for (var j:int=0; j<4; j++) {
                for (var i:int=0; i<4; i++) {
                    ba.writeFloat(matrix.m[j + i*4]);
                }
            }
            return ba;
        }
        
    }
}
