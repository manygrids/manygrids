package me.ylab.graphic.types.indices.textured {
    
    import flash.display3D.Context3D;
    import flash.display3D.Context3DProgramType;
    import flash.display3D.Context3DVertexBufferFormat;
    import flash.display3D.IndexBuffer3D;
    import flash.display3D.VertexBuffer3D;
    
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.tools.MyCoordinateSystem3;
    import me.ylab.graphic.types.indices.MyVertexIndicesMesh;
    import me.ylab.graphic.types.indices.MyVertexIndicesMeshDraw;
    
    public class MyTexturedVertexIndicesMeshDraw extends MyVertexIndicesMeshDraw {
        
        public function MyTexturedVertexIndicesMeshDraw(mesh:MyVertexIndicesMesh, vb:VertexBuffer3D, ib:IndexBuffer3D) {
            super(mesh, vb, ib);
        }
        
        public override function draw():void {
            var context3D:Context3D = mGraphic.context;
            // vertex position to attribute register 0
            context3D.setVertexBufferAt(0, mFlashVertexBuffer, 0, Context3DVertexBufferFormat.FLOAT_3);
            // UV to attribute register 1
            context3D.setVertexBufferAt(1, mFlashVertexBuffer, 3, Context3DVertexBufferFormat.FLOAT_2);
            
            mDrawMatrix.resetBy(mCameraMatrix);
            mDrawMatrix.multiply(mFatherDrawMatrix);
            mDrawMatrix.multiply(mMesh.coordinate.selfToWorld);
            mDrawMatrix.fillTransposeByteArray(mDrawMatrixByteArray);
            
            context3D.setProgramConstantsFromByteArray(Context3DProgramType.VERTEX, 0, 4, mDrawMatrixByteArray, 0);
            
            context3D.drawTriangles(mFlashIndexBuffer);
            context3D.setVertexBufferAt(1, null);
        }
    }
}
