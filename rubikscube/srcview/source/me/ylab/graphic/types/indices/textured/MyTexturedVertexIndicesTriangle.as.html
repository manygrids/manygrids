package me.ylab.graphic.types.indices.textured {
    import me.ylab.graphic.foundation.MyVertexBuffer;
    import me.ylab.graphic.types.TexturedVertex3;
    import me.ylab.graphic.types.indices.MyVertexIndicesTriangle;
    import me.ylab.tools.MyTriangle;
    
    public class MyTexturedVertexIndicesTriangle extends MyVertexIndicesTriangle {
        
        public function MyTexturedVertexIndicesTriangle(p0:int, p1:int, p2:int,
                                                       vertexBuffer:MyVertexBuffer = null,
                                                       perTriangleByte:int = 0) {
            super(p0, p1, p2, vertexBuffer, perTriangleByte);
            mVertexType = new TexturedVertex3();
            if (null != mVertexBuffer) {
                mVertexType.getFromVertexBuffer(mVertexBuffer, mIndexP0);
                mCross.setXYZ(mVertexType.x, mVertexType.y, mVertexType.z);
                mVertexType.getFromVertexBuffer(mVertexBuffer, mIndexP1);
                mPoint1.setXYZ(mVertexType.x, mVertexType.y, mVertexType.z);
                mVertexType.getFromVertexBuffer(mVertexBuffer, mIndexP2);
                mPoint2.setXYZ(mVertexType.x, mVertexType.y, mVertexType.z);
                refreshNormal();
            }
            
        }
        
        public static function CreateFrom(tri:MyTexturedVertexIndicesTriangle):MyTriangle {
            var ret:MyTexturedVertexIndicesTriangle = 
                new MyTexturedVertexIndicesTriangle(tri.mIndexP0, tri.mIndexP1, tri.mIndexP2,
                                                    tri.mVertexBuffer, tri.mPerTriangleByte);
            return ret;
        }
        

    }
}
