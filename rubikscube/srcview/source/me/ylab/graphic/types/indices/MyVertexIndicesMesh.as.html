package me.ylab.graphic.types.indices {
    import flash.display3D.Context3D;
    import flash.display3D.Context3DProgramType;
    import flash.display3D.Context3DVertexBufferFormat;
    import flash.display3D.IndexBuffer3D;
    import flash.display3D.VertexBuffer3D;
    import flash.geom.Matrix3D;
    import flash.utils.ByteArray;
    import flash.utils.Endian;
    
    import me.ylab.graphic.foundation.EmptyDraw;
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.foundation.IDrawable;
    import me.ylab.graphic.foundation.ISelectable;
    import me.ylab.graphic.foundation.MyColor4f;
    import me.ylab.graphic.foundation.MyDrawBatch;
    import me.ylab.graphic.foundation.MyDrawBatchPackage;
    import me.ylab.graphic.foundation.MyIndexBuffer;
    import me.ylab.graphic.foundation.MyMesh;
    import me.ylab.graphic.foundation.MyVertexBuffer;
    import me.ylab.tools.MyCoordinateSystem3;
    import me.ylab.tools.MyTriangle;

    public class MyVertexIndicesMesh extends MyMesh implements ISelectable {

        protected var mVertexBuffer:MyVertexBuffer;
        protected var mIndexBuffer:MyIndexBuffer;
        
        protected var mFlashVertexBuffer:VertexBuffer3D;
        protected var mFlashIndexBuffer:IndexBuffer3D;
        protected var mDraw:IDrawable;
        protected var mColorByteArray:ByteArray;
        protected var mIsSelected:Boolean;
        protected var mSelectedColor:MyColor4f;
        protected var mOriginColor:MyColor4f;
        
        public function MyVertexIndicesMesh(graphic:Graphic3D = null,
                                            color:MyColor4f = null,
                                            vertexBuffer:MyVertexBuffer = null,
                                            indexBuffer:MyIndexBuffer = null) {
            super(graphic, color);
            mOriginColor = new MyColor4f(mColor.r, mColor.g, mColor.b, mColor.a);
            mSelectedColor = new MyColor4f;
            mVertexBuffer = vertexBuffer;
            mIndexBuffer = indexBuffer;
            mIsSelected = false;
            
            if (null == mVertexBuffer) mVertexBuffer = new MyVertexBuffer();
            if (null == mIndexBuffer) mIndexBuffer = new MyIndexBuffer();
            mDraw = EmptyDraw.instance;

            mColorByteArray = new ByteArray;
            mColorByteArray.endian = Endian.LITTLE_ENDIAN;
            mColorByteArray.writeFloat(mColor.r);
            mColorByteArray.writeFloat(mColor.g);
            mColorByteArray.writeFloat(mColor.b);
            mColorByteArray.writeFloat(mColor.a);
        }
        
        public static function CreateFrom(mesh:MyVertexIndicesMesh):MyVertexIndicesMesh {
            var ret:MyVertexIndicesMesh = new MyVertexIndicesMesh(mesh.graphic, mesh.color, mesh.mVertexBuffer, mesh.mIndexBuffer);
            ret.mFlashIndexBuffer = mesh.mFlashIndexBuffer;
            ret.mFlashVertexBuffer = mesh.mFlashVertexBuffer;
            ret.mModel = mesh.mModel;
            ret.mPrepared = false;
            ret.mCoordinate.selfToWorld.resetBy(mesh.mCoordinate.selfToWorld);
            ret.mCoordinate.worldToSelf.resetBy(mesh.mCoordinate.worldToSelf);
            for (var i:int=0; i<mesh.mTriangles.length; i++) {
                ret.mTriangles.push(MyVertexIndicesTriangle.CreateFrom(MyVertexIndicesTriangle(mesh.mTriangles[i])));
            }
            return ret;
        }
        
        public function get vertexBuffer():MyVertexBuffer {
            return mVertexBuffer;
        }
        
        public override function addTriangle(tri:MyTriangle):void {
            mTriangles.push(tri);
            mIndexBuffer.addTriangle(MyVertexIndicesTriangle(tri));
            MyVertexIndicesTriangle(tri).vertexBuffer = mVertexBuffer;
        }
        
        public override function setColor(r:Number, g:Number, b:Number, a:Number):void {
            mColor.setRGBA(r, g, b, a);
            mColorByteArray.clear();
            mColorByteArray.writeFloat(r);
            mColorByteArray.writeFloat(g);
            mColorByteArray.writeFloat(b);
            mColorByteArray.writeFloat(a);
        }
        
        public function get selectedColor():MyColor4f {
            return mSelectedColor;
        }
        
        public function set selectedColor(color:MyColor4f):void {
            mSelectedColor = color;
        }
        
        public function get colorByteArray():ByteArray {
            return mColorByteArray;
        }
        
        public override function set graphic(g:Graphic3D):void {
            mGraphic = g;
        }

        public function bindVertexBuffer():void {
            var context3D:Context3D = mGraphic.context;
            // vertex position to attribute register 0
            context3D.setVertexBufferAt(0, mFlashVertexBuffer, 0, Context3DVertexBufferFormat.FLOAT_3);
        }
        
        public function createDrawBuffers():void {
            var context3D:Context3D = mGraphic.context;
            if (null != mFlashVertexBuffer) {
                mFlashVertexBuffer.dispose();
            }
            mFlashVertexBuffer = context3D.createVertexBuffer(mVertexBuffer.count, mVertexBuffer.numberVector.length / mVertexBuffer.count);
            mFlashVertexBuffer.uploadFromByteArray(mVertexBuffer.byteArray, 0, 0, mVertexBuffer.count);
            
            if (null != mFlashIndexBuffer) {
                mFlashIndexBuffer.dispose();
            }
            mFlashIndexBuffer = context3D.createIndexBuffer(mIndexBuffer.count);
            mFlashIndexBuffer.uploadFromByteArray(mIndexBuffer.byteArray, 0, 0, mIndexBuffer.count);
        }
        
        public override function prepareDraw():void {
            if (mPrepared || null == mGraphic) return;
            mPrepared = true;

            if (mDraw == EmptyDraw.instance) {
                mDraw = new MyVertexIndicesMeshDraw(this, mFlashVertexBuffer, mFlashIndexBuffer);
            }
            
            var viDrawBatch:MyDrawBatch = MyVertexIndicesDrawBatch.getDrawBatch(mGraphic, true);
            var drawPkg:MyDrawBatchPackage = new MyDrawBatchPackage(mGraphic, viDrawBatch, mModel, mDraw);
            
            viDrawBatch.addDrawBatchPackage(drawPkg, mIsTransparent);
        }
        
        public override function unprepare():void {
            if (!mPrepared || null == mGraphic) return;
            mPrepared = false;
            
            var viDrawBatch:MyDrawBatch = MyVertexIndicesDrawBatch.getDrawBatch(mGraphic, true);
            var drawPkg:MyDrawBatchPackage = new MyDrawBatchPackage(mGraphic, viDrawBatch, mModel, mDraw);
            
            viDrawBatch.removeDrawBatchPackage(drawPkg);
        }

        public override function draw():void {
            mDraw.draw();
        }
        
        public function isSelected():Boolean {
            return mIsSelected;
        }
        
        public function select():void {
            if (mIsSelected) return;
            mIsSelected = true;
            mOriginColor.setRGBA(mColor.r, mColor.g, mColor.b, mColor.a);
            setColor(mSelectedColor.r, mSelectedColor.g, mSelectedColor.b, mSelectedColor.a);
        }
        
        public function unselect():void {
            if (!mIsSelected) return;
            mIsSelected = false;
            setColor(mOriginColor.r, mOriginColor.g, mOriginColor.b, mOriginColor.a);
        }
        
    }
}
