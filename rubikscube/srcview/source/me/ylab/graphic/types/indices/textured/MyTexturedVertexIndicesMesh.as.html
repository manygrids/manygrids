package me.ylab.graphic.types.indices.textured {
    
    import flash.display3D.Context3D;
    
    import me.ylab.graphic.foundation.EmptyDraw;
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.foundation.MyDrawBatch;
    import me.ylab.graphic.foundation.MyDrawBatchPackage;
    import me.ylab.graphic.foundation.MyIndexBuffer;
    import me.ylab.graphic.foundation.MyTexture;
    import me.ylab.graphic.foundation.MyVertexBuffer;
    import me.ylab.graphic.types.TexturedVertex3;
    import me.ylab.graphic.types.indices.MyVertexIndicesMesh;
    import me.ylab.tools.MyTriangle;
    import me.ylab.tools.MyVertex3;
    
    public class MyTexturedVertexIndicesMesh extends MyVertexIndicesMesh {
        
        protected var mTexture:MyTexture;
        
        public function MyTexturedVertexIndicesMesh(graphic:Graphic3D=null, vertexBuffer:MyVertexBuffer=null, indexBuffer:MyIndexBuffer=null,
                                                    texture:MyTexture=null) {
            super(graphic, null, vertexBuffer, indexBuffer);
            mIsTransparent = true;
            mTexture = texture;
            if (null != mTexture) {
                var tHalfWidth:Number = mTexture.width * mTexture.widthRatio * 0.5 * 0.01;
                var tHalfHeight:Number = mTexture.height * mTexture.heightRatio * 0.5 * 0.01;
                
                mVertexBuffer.addGraphicVertex(new TexturedVertex3(-tHalfWidth, -tHalfHeight, 0, 0, mTexture.heightRatio));
                mVertexBuffer.addGraphicVertex(new TexturedVertex3( tHalfWidth, -tHalfHeight, 0, mTexture.widthRatio, mTexture.heightRatio));
                mVertexBuffer.addGraphicVertex(new TexturedVertex3(-tHalfWidth,  tHalfHeight, 0, 0, 0));
                mVertexBuffer.addGraphicVertex(new TexturedVertex3( tHalfWidth,  tHalfHeight, 0, mTexture.widthRatio, 0));
                
                addTriangle(new MyTexturedVertexIndicesTriangle(0, 1, 2));
                addTriangle(new MyTexturedVertexIndicesTriangle(2, 1, 3));
            }
        }
        
        public function get texture():MyTexture {
            return mTexture;
        }
        
        public function set texture(t:MyTexture):void {
            if (mPrepared) {
                unprepare();
                mTexture = t;
                prepareDraw();
                return;
            }
            mTexture = t;
        }

        public override function addTriangle(tri:MyTriangle):void {
            mTriangles.push(tri);
            mIndexBuffer.addTriangle(MyTexturedVertexIndicesTriangle(tri));
            MyTexturedVertexIndicesTriangle(tri).vertexBuffer = mVertexBuffer;
        }
        
        public override function prepareDraw():void {
            if (mPrepared || null == mGraphic) return;
            mPrepared = true;
            
            var context3D:Context3D = mGraphic.context;
            if (null == mFlashVertexBuffer) {
                mFlashVertexBuffer = context3D.createVertexBuffer(mVertexBuffer.count, mVertexBuffer.numberVector.length / mVertexBuffer.count);
                mFlashVertexBuffer.uploadFromByteArray(mVertexBuffer.byteArray, 0, 0, mVertexBuffer.count);
            }
            
            if (null == mFlashIndexBuffer) {
                mFlashIndexBuffer = context3D.createIndexBuffer(mIndexBuffer.count);
                mFlashIndexBuffer.uploadFromByteArray(mIndexBuffer.byteArray, 0, 0, mIndexBuffer.count);
            }
            
            if (mDraw == EmptyDraw.instance) {
                mDraw = new MyTexturedVertexIndicesMeshDraw(this, mFlashVertexBuffer, mFlashIndexBuffer);
            }
            
            var viDrawBatch:MyDrawBatch = MyTexturedVertexIndicesDrawBatch.getDrawBatch(mGraphic, mTexture, true);
            var drawPkg:MyDrawBatchPackage = new MyDrawBatchPackage(mGraphic, viDrawBatch, mModel, mDraw);
            
            viDrawBatch.addDrawBatchPackage(drawPkg, mIsTransparent);
        }
        
        public override function unprepare():void {
            if (!mPrepared || null == mGraphic) return;
            mPrepared = false;
            
            var viDrawBatch:MyDrawBatch = MyTexturedVertexIndicesDrawBatch.getDrawBatch(mGraphic, mTexture, true);
            var drawPkg:MyDrawBatchPackage = new MyDrawBatchPackage(mGraphic, viDrawBatch, mModel, mDraw);
            
            viDrawBatch.removeDrawBatchPackage(drawPkg);
        }
        
        public static function CreateFrom(mesh:MyTexturedVertexIndicesMesh):MyTexturedVertexIndicesMesh {
            var ret:MyTexturedVertexIndicesMesh = new MyTexturedVertexIndicesMesh(mesh.graphic, mesh.mVertexBuffer, mesh.mIndexBuffer, mesh.mTexture);
            ret.mModel = mesh.mModel;
            ret.mPrepared = false;
            ret.mCoordinate.selfToWorld.resetBy(mesh.mCoordinate.selfToWorld);
            ret.mCoordinate.worldToSelf.resetBy(mesh.mCoordinate.worldToSelf);
            for (var i:int=0; i<mesh.mTriangles.length; i++) {
                ret.mTriangles.push(MyTexturedVertexIndicesTriangle.CreateFrom(MyTexturedVertexIndicesTriangle(mesh.mTriangles[i])));
            }
            return ret;
        }
        
    }
}
