package me.ylab.graphic.types.indices.cubetexture {
    
    import flash.utils.Dictionary;
    
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.foundation.MyCubeTexture;
    import me.ylab.graphic.foundation.MyDrawBatch;
    
    public class MyVertexIndicesWithCubeTextureDrawBatch extends MyDrawBatch {
        
        protected static var mDrawBatches:Dictionary = new Dictionary;
        protected var mTexture:MyCubeTexture;
        
        public function MyVertexIndicesWithCubeTextureDrawBatch(graphic:Graphic3D, texture:MyCubeTexture) {
            super(graphic);
            mTexture = texture;
        }
        
        public static function getDrawBatch(graphic:Graphic3D, texture:MyCubeTexture, autoCreate:Boolean = false):MyDrawBatch {
            var dbs:Dictionary = mDrawBatches[graphic];
            if (null == dbs) {
                if (autoCreate) {
                    dbs = new Dictionary;
                    mDrawBatches[graphic] = dbs;
                } else {
                    return null;
                }
            }
            
            var db:MyDrawBatch = dbs[texture];
            if (null == db && autoCreate) {
                db = new MyVertexIndicesWithCubeTextureDrawBatch(graphic, texture);
                dbs[texture] = db;
            }
            
            return db;
        }
        
        public override function draw():void {
            mGraphic.context.setProgram(mGraphic.program.vertexIndicesWithCubeTextureProgram());
            mGraphic.context.setTextureAt(0, mTexture.texture);
            super.draw();
            mGraphic.context.setTextureAt(0, null);
        }
    }
}
