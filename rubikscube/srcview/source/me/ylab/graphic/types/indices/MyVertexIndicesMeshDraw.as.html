package me.ylab.graphic.types.indices {

    import flash.display3D.Context3D;
    import flash.display3D.Context3DProgramType;
    import flash.display3D.Context3DTriangleFace;
    import flash.display3D.Context3DVertexBufferFormat;
    import flash.display3D.IndexBuffer3D;
    import flash.display3D.VertexBuffer3D;
    import flash.utils.ByteArray;
    import flash.utils.Endian;
    
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.foundation.IDrawable;
    import me.ylab.graphic.foundation.MyMesh;
    import me.ylab.graphic.types.GraphicMatrix4x4;
    import me.ylab.tools.MyCoordinateSystem3;
    import me.ylab.tools.MyMatrix4x4;
    
    public class MyVertexIndicesMeshDraw implements IDrawable {

        protected var mGraphic:Graphic3D;
        protected var mFlashVertexBuffer:VertexBuffer3D;
        protected var mFlashIndexBuffer:IndexBuffer3D;
        protected var mDrawMatrix:GraphicMatrix4x4;
        protected var mDrawMatrixByteArray:ByteArray;
        protected var mSelfMatrixByteArray:ByteArray;
        protected var mFatherDrawMatrix:GraphicMatrix4x4;
        protected var mCameraMatrix:GraphicMatrix4x4;
        protected var mMesh:MyVertexIndicesMesh;
        
        public function MyVertexIndicesMeshDraw(mesh:MyVertexIndicesMesh, vb:VertexBuffer3D, ib:IndexBuffer3D) {
            mMesh = mesh;
            mGraphic = mesh.graphic;
            mFlashVertexBuffer = vb;
            mFlashIndexBuffer = ib;
            mDrawMatrix = new GraphicMatrix4x4;
            mDrawMatrixByteArray = new ByteArray;
            mDrawMatrixByteArray.endian = Endian.LITTLE_ENDIAN;
            mSelfMatrixByteArray = new ByteArray;
            mSelfMatrixByteArray.endian = Endian.LITTLE_ENDIAN;
            GraphicMatrix4x4.FillTransposeByteArray(mSelfMatrixByteArray, mMesh.coordinate.selfToWorld);
        }
        
        public function set vertexBuffer(vb:VertexBuffer3D):void {
            mFlashVertexBuffer = vb;
        }
        
        public function set indexBuffer(ib:IndexBuffer3D):void {
            mFlashIndexBuffer = ib;
        }
        
        public function draw():void {
            var context3D:Context3D = mGraphic.context;
            // vertex position to attribute register 0
//            context3D.setVertexBufferAt(0, mFlashVertexBuffer, 0, Context3DVertexBufferFormat.FLOAT_3);
            
            GraphicMatrix4x4.FillTransposeByteArray(mDrawMatrixByteArray, mFatherDrawMatrix);

            context3D.setProgramConstantsFromByteArray(Context3DProgramType.VERTEX, 0, 4, mDrawMatrixByteArray, 0);
            context3D.setProgramConstantsFromByteArray(Context3DProgramType.VERTEX, 4, 4, mSelfMatrixByteArray, 0);
            // set color
            context3D.setProgramConstantsFromByteArray(Context3DProgramType.FRAGMENT, 0, 1, mMesh.colorByteArray, 0);
            
            context3D.drawTriangles(mFlashIndexBuffer);
        }
        
        public function getDrawMatrix():GraphicMatrix4x4 {
            return mFatherDrawMatrix;
        }
        
        public function setDrawMatrix(matrix:GraphicMatrix4x4):void {
            mFatherDrawMatrix = matrix;
        }
        
        public function setCameraMatrix(matrix:GraphicMatrix4x4):void {
            mCameraMatrix = matrix;
        }
        
    }
}
