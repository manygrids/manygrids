package me.ylab.graphic.types.indices.cubetexture {

    import flash.display3D.Context3D;
    import flash.display3D.Context3DProgramType;
    import flash.display3D.Context3DVertexBufferFormat;
    import flash.display3D.IndexBuffer3D;
    import flash.display3D.VertexBuffer3D;
    import flash.utils.ByteArray;
    import flash.utils.Endian;
    
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.types.GraphicMatrix4x4;
    import me.ylab.graphic.types.indices.MyVertexIndicesMesh;
    import me.ylab.graphic.types.indices.MyVertexIndicesMeshDraw;
    import me.ylab.tools.MyCoordinateSystem3;
    import me.ylab.tools.MyMatrix4x4;
    import me.ylab.tools.MyVector4;
    import me.ylab.tools.MyVertex4;
    
    public class MyVertexIndicesWithCubeTextureMeshDraw extends MyVertexIndicesMeshDraw {

        protected var mNormalByteArray:ByteArray;
        protected var mEyePosByteArray:ByteArray;
        protected var mTextureRateByteArray:ByteArray;
        protected var mWorldMatrix:GraphicMatrix4x4;
        protected var mWorldMatrixByteArray:ByteArray;
        protected var mCameraMatrixByteArray:ByteArray;

        public function MyVertexIndicesWithCubeTextureMeshDraw(mesh:MyVertexIndicesWithCubeTextureMesh, vb:VertexBuffer3D, ib:IndexBuffer3D) {
            super(mesh, vb, ib);
            mNormalByteArray = new ByteArray;
            mNormalByteArray.endian = Endian.LITTLE_ENDIAN;
            mEyePosByteArray = new ByteArray;
            mEyePosByteArray.endian = Endian.LITTLE_ENDIAN;
            mTextureRateByteArray = new ByteArray;
            mTextureRateByteArray.endian = Endian.LITTLE_ENDIAN;
            var rate:Number = 0.3;
            mTextureRateByteArray.writeFloat(rate);
            mTextureRateByteArray.writeFloat(rate);
            mTextureRateByteArray.writeFloat(rate);
            mTextureRateByteArray.writeFloat(rate);
            
            mWorldMatrix = new GraphicMatrix4x4;
            mWorldMatrixByteArray = new ByteArray;
            mWorldMatrixByteArray.endian = Endian.LITTLE_ENDIAN;
            mCameraMatrixByteArray = new ByteArray;
            mCameraMatrixByteArray.endian = Endian.LITTLE_ENDIAN;
        }
        
        public override function draw():void {
            var context3D:Context3D = mGraphic.context;
            // vertex position to attribute register 0
            context3D.setVertexBufferAt(0, mFlashVertexBuffer, 0, Context3DVertexBufferFormat.FLOAT_3);
            
            mWorldMatrix.resetBy(mGraphic.currentCamera.viewMatrix);
            mWorldMatrix.m[12] = mWorldMatrix.m[13] = mWorldMatrix.m[14] = 0;
            mWorldMatrix.multiply(mFatherDrawMatrix);
            mWorldMatrix.multiply(mMesh.coordinate.selfToWorld);
            mWorldMatrix.fillTransposeByteArray(mWorldMatrixByteArray);

            mDrawMatrix.resetBy(mCameraMatrix);
            mDrawMatrix.multiply(mFatherDrawMatrix);
            mDrawMatrix.multiply(mMesh.coordinate.selfToWorld);
            mDrawMatrix.fillTransposeByteArray(mDrawMatrixByteArray);
            
            var ori:MyVertex4 = new MyVertex4;
            ori = MyMatrix4x4.MultiplyVertex(mWorldMatrix, ori);
            var normal:MyVertex4 = MyMatrix4x4.MultiplyVertex(mWorldMatrix, mMesh.normal);
            var nor:MyVector4 = MyVector4.Minus(normal, ori);
            mNormalByteArray.clear();
            mNormalByteArray.writeFloat(nor.x);
            mNormalByteArray.writeFloat(nor.y);
            mNormalByteArray.writeFloat(nor.z);
            mNormalByteArray.writeFloat(1);

            mEyePosByteArray.clear();
            mEyePosByteArray.writeFloat(0);
            mEyePosByteArray.writeFloat(0);
            mEyePosByteArray.writeFloat(-mGraphic.currentCamera.eyeOn.z);
            mEyePosByteArray.writeFloat(1);

            context3D.setProgramConstantsFromByteArray(Context3DProgramType.VERTEX, 0, 4, mDrawMatrixByteArray, 0);
            context3D.setProgramConstantsFromByteArray(Context3DProgramType.VERTEX, 4, 4, mWorldMatrixByteArray, 0);
            context3D.setProgramConstantsFromByteArray(Context3DProgramType.VERTEX, 8, 1, mNormalByteArray, 0);
            context3D.setProgramConstantsFromByteArray(Context3DProgramType.VERTEX, 9, 1, mEyePosByteArray, 0);
            // multiply component
            context3D.setProgramConstantsFromByteArray(Context3DProgramType.FRAGMENT, 0, 1, mTextureRateByteArray, 0);
            // set color
            context3D.setProgramConstantsFromByteArray(Context3DProgramType.FRAGMENT, 1, 1, mMesh.colorByteArray, 0);

            context3D.drawTriangles(mFlashIndexBuffer);
            context3D.setVertexBufferAt(1, null);
        }
    }
}
