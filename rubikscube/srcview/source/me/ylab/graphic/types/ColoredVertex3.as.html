package me.ylab.graphic.types {
    import me.ylab.graphic.foundation.MyVertexBuffer;
    import me.ylab.graphic.foundation.MyColor4f;

    public class ColoredVertex3 extends GraphicVertex3 {
        
        protected var mColor:MyColor4f;

        public function ColoredVertex3(x:Number = 0, y:Number = 0, z:Number = 0,
                                       r:Number = 1, g:Number = 1, b:Number = 1, a:Number = 1) {
            super(x, y, z);
            mColor = new MyColor4f(r, g, b, a);
        }
        
        public function get color():MyColor4f {
            return mColor;
        }

        public override function addTo(v:Vector.<Number>):void {
            v.push(mX);
            v.push(mY);
            v.push(mZ);
            v.push(mColor.r);
            v.push(mColor.g);
            v.push(mColor.b);
            v.push(mColor.a);
        }
        
        public override function getFromVertexBuffer(vb:MyVertexBuffer, index:int):void {
            mX = vb.numberVector[index*7];
            mY = vb.numberVector[index*7 + 1];
            mZ = vb.numberVector[index*7 + 2];
            mColor.setRGBA(vb.numberVector[index*7 + 3], vb.numberVector[index*7 + 4], vb.numberVector[index*7 + 5], vb.numberVector[index*7 + 6]);
        }

    }
}
