package me.ylab.graphic.foundation {
    import me.ylab.tools.MyTriangle;
    import me.ylab.tools.MyVertex3;
    import me.ylab.tools.MyVertex4;

    public class CrossTestResult {

        public var result:Boolean;
        public var tri:MyTriangle;
        public var vertex:MyVertex4;
        public var mesh:MyMesh;
        public var subModel:MyModel;
        public var model:MyModel;
        
        public function CrossTestResult(r:Boolean = false,
                                        t:MyTriangle = null,
                                        v:MyVertex4 = null,
                                        me:MyMesh = null,
                                        mo:MyModel = null,
                                        sub:MyModel = null) {
            result = r;
            tri = t;
            vertex = v;
            mesh = me;
            subModel = sub;
            model = mo;
        }
        
        public function resetBy(ctResult:CrossTestResult):void {
            result = ctResult.result;
            tri = ctResult.tri;
            vertex = ctResult.vertex;
            mesh = ctResult.mesh;
            subModel = ctResult.subModel;
            model = ctResult.model;
        }
    }
}
