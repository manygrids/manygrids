        public override function fillDrawMatrix(matrix:MyMatrix4x4):void {
            matrix.identify();
            matrix.multiply(mCoordinate.selfToWorld);
        }

        public override function obtainWorldCoordinate():MyCoordinateSystem3 {

        }

            var drawable:MyDrawBatchPackage;

//                drawable.drawBatch.setCameraMatrix(mCameraMatrix);
                drawable.drawBatch.setDrawMatrix(mDrawMatrix);
                drawable.drawBatch.draw();
