package me.ylab.graphic.foundation {
    
    import flash.display.Stage;
    import flash.display.Stage3D;
    import flash.display3D.Context3D;
    import flash.display3D.Context3DBlendFactor;
    import flash.display3D.Context3DClearMask;
    import flash.display3D.Context3DCompareMode;
    import flash.display3D.Context3DProgramType;
    import flash.display3D.Context3DRenderMode;
    import flash.display3D.Context3DStencilAction;
    import flash.display3D.Context3DTriangleFace;
    import flash.display3D.Context3DVertexBufferFormat;
    import flash.display3D.Program3D;
    import flash.events.Event;
    import flash.text.engine.ContentElement;
    import flash.utils.getTimer;
    
    import me.ylab.tools.MyCoordinateSystem3;
    import me.ylab.tools.MyRay3;
    import me.ylab.tools.MyVector3;
    import me.ylab.tools.MyVertex3;

    public class Graphic3D {

        public static var INIT_FINISHED:int = 1;
        public static var KEY_EVENT_DOWN:int = 10;
        
        protected var mEvents:Array;
        protected var mStage:Stage;
        protected var mStage3D:Stage3D;
        protected var mContext3D:Context3D;
        protected var mWidth:Number;
        protected var mHeight:Number;
        protected var mClearColor:MyColor4f;
        protected var mProgram:MyProgram;
        protected var mCurrentCamera:MyCamera3D;
        protected var mWorld:MyModel;
        protected var mFPS:Number;
        protected var mFrames:int;
        protected var mLastTime:int;
        protected var mLapse:Number;
        
        private var mCrossTestRes:CrossTestResult;
        private var mSelectState:Boolean;
        private var mSelectX:int;
        private var mSelectY:int;
        private var mCallTimes:int;
        
        private var mMousePosX:int;
        private var mMousePosY:int;
        private var mLastMousePosX:int;
        private var mLastMousePosY:int;
        
        public function Graphic3D(stage:Stage, width:Number = 100, height:Number = 100) {
            mEvents = new Array();
            mLastTime = 0;
            mFrames = 0;
            mFPS = 0;
            mWidth = width;
            mHeight = height;
            mClearColor = new MyColor4f(1, 1, 1, 1);
            mWorld = new WorldModel(this);
            
            mSelectState = false;
            mSelectX = 0;
            mSelectY = 0;
            mCrossTestRes = new CrossTestResult;
            mCallTimes = 0;
            
            mCurrentCamera = new MyCamera3D;
            mCurrentCamera.setPerspectiveByFovY(Math.PI / 2, 1);
            mCurrentCamera.setEyeParam(new MyVertex3(0, 0, 10), new MyVertex3(0, 0, 0), new MyVector3(0, 1, 0));

            mStage = stage;
            mStage3D = stage.stage3Ds[0];
            mStage3D.addEventListener(Event.CONTEXT3D_CREATE, init3D);
            mStage3D.requestContext3D(Context3DRenderMode.AUTO);
        }
        
        public function init3D(e:Event):void {
            mContext3D = mStage3D.context3D;
            mContext3D.configureBackBuffer(mWidth, mHeight, 0, true);
            mContext3D.setCulling(Context3DTriangleFace.FRONT);
            mContext3D.setBlendFactors(Context3DBlendFactor.SOURCE_ALPHA, Context3DBlendFactor.ONE_MINUS_SOURCE_ALPHA);

            mProgram = new MyProgram(this);

            callEvent(INIT_FINISHED);
        }

        public function get selectState():Boolean {
            return mSelectState;
        }
        
        public function set selectState(state:Boolean):void {
            mSelectState = state;
        }
        
        public function set selectX(x:int):void {
            mSelectX = x;
        }
        
        public function set selectY(y:int):void {
            mSelectY = y;
        }
        
        public function get selectResult():CrossTestResult {
            var tRes:CrossTestResult = testCrossByRay(mCurrentCamera.getEyeRayByScreenPoint(mSelectX, mSelectY));
            return tRes;
        }
        
        public function set mousePosX(x:int):void {
            mMousePosX = x;
        }
        
        public function set mousePosY(y:int):void {
            mMousePosY = y;
        }
        
        public function setClearColor(r:Number, g:Number, b:Number, a:Number):void {
            mClearColor.setRGBA(r, g, b, a);
        }
        
        public function addEventListener(e:int, fun:Function):void {
            if (null == mEvents[e]) mEvents[e] = new Array;
            mEvents[e].push(fun);
        }

        public function useCamera(camera:MyCamera3D):void {
            mCurrentCamera = camera;
        }
        
        public function get currentCamera():MyCamera3D {
            return mCurrentCamera;
        }
        
        public function resetCurrentCamera():void {
            mCurrentCamera.setEyeParam(new MyVertex3(0, 0, 10), new MyVertex3(0, 0, 0), new MyVector3(0, 1, 0));
        }
        
        protected function callEvent(e:int):void {
            if (null == mEvents[e]) return;
            for each (var f:Function in mEvents[e]) {
                f.call();
            }
        }

        public function get world():MyModel {
            return mWorld;
        }
        
        public function get fps():Number {
            return mFPS;
        }

        public function graphicLogic():void {
            if (mSelectState) {
                var tRes:CrossTestResult = testCrossByRay(mCurrentCamera.getEyeRayByScreenPoint(mSelectX, mSelectY));
                mCrossTestRes.result = tRes.result;
                mCrossTestRes.tri = tRes.tri;
                mCrossTestRes.vertex = tRes.vertex;
                mCrossTestRes.mesh = tRes.mesh;
                mCrossTestRes.model = tRes.model;
                mCrossTestRes.subModel = tRes.subModel;
            }
        }
        
        public function onRender():void {
            mLapse = getTimer() - mLastTime;
            mFrames++;
            if (mLapse > 1000) {
                mFPS = 1000 * mFrames / mLapse;
                mLastTime = getTimer();
                mFrames = 0;
            }

            mContext3D.clear(mClearColor.r, mClearColor.g, mClearColor.b, mClearColor.a);
            
            mWorld.render();
            
            mContext3D.present();
        }
        
        public function get program():MyProgram {
            return mProgram;
        }
        
        public function get context():Context3D {
            return mContext3D;
        }
        
        public function addModel(model:MyModel):void {
            mWorld.addModel(model);
            model.prepareRender();
        }
        
        public function removeModel(model:MyModel):void {
            model.unprepare();
            mWorld.removeModel(model);
        }
        
        protected function testCrossByRay(ray:MyRay3):CrossTestResult {
            return mWorld.testCrossByRay(ray);
        }
        
    }
}
