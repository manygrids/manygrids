package me.ylab.graphic.foundation {
    import flash.display.Bitmap;
    import flash.display.BitmapData;
    import flash.display.Loader;
    import flash.display.LoaderInfo;
    import flash.display3D.Context3DTextureFormat;
    import flash.display3D.textures.Texture;
    import flash.events.Event;
    import flash.events.IOErrorEvent;
    import flash.events.ProgressEvent;
    import flash.net.FileReference;
    import flash.net.URLLoader;
    import flash.net.URLRequest;
    import flash.net.URLStream;
    import flash.utils.ByteArray;
    import flash.utils.Endian;
    
    import me.ylab.application.IResource;
    import me.ylab.application.MyResource;

    public class MyTexture implements IResource {

        protected var mGraphic:Graphic3D;
        protected var mReadError:Boolean;
        protected var mTexture:Texture;
        protected var mReadingPos:uint;
        protected var mFileByteArray:ByteArray;
        protected var mUserCount:int;
        protected var mFileName:String;
        protected var mIsLoaded:Boolean;
        protected var mEventListeners:Vector.<Vector.<Function>>;
        
        protected var mWidth:int;
        protected var mHeight:int;
        
        protected var mWidthRatio:Number;
        protected var mHeightRatio:Number;
        
        public function MyTexture(graphic:Graphic3D) {
            mGraphic = graphic;
            mFileByteArray = new ByteArray;
            mEventListeners = new Vector.<Vector.<Function>>;
            mEventListeners[MyResource.LOAD_COMPLETE] = new Vector.<Function>;
        }
        
        public function clear():void {
            var transparentBD:BitmapData = new BitmapData(mWidth, mHeight, true, 0);
            mWidthRatio = 1;
            mHeightRatio = 1;
            mTexture.uploadFromBitmapData(transparentBD);
            transparentBD.dispose();
        }
        
        private function onDecodeComplete(e:Event):void {
            var width:int = Bitmap(LoaderInfo(e.target).content).bitmapData.width;
            var height:int = Bitmap(LoaderInfo(e.target).content).bitmapData.height;
            
            mWidth = 1;
            mHeight = 1;
            
            while (mWidth < width) mWidth *= 2;
            while (mHeight < height) mHeight *= 2;
            
            mWidthRatio = Number(width) / mWidth;
            mHeightRatio = Number(height) / mHeight;
            
            mTexture = mGraphic.context.createTexture(mWidth, mHeight, Context3DTextureFormat.BGRA, false);
            
            mTexture.uploadFromBitmapData(Bitmap(LoaderInfo(e.target).content).bitmapData);
            
            mIsLoaded = true;
            for each (var f:Function in mEventListeners[MyResource.LOAD_COMPLETE]) {
                f.call(null, this);
            }
        }
        
        public function get width():int {
            return mWidth;
        }
        
        public function get height():int {
            return mHeight;
        }
        
        public function get widthRatio():Number {
            return mWidthRatio;
        }
        
        public function get heightRatio():Number {
            return mHeightRatio;
        }
        
        private function onReading(e:Event):void {
            var file:URLStream = URLStream(e.target);
            var availableBytes:uint = file.bytesAvailable;
            if (availableBytes > 0) {
                file.readBytes(mFileByteArray, mReadingPos);
                mReadingPos += availableBytes;
            }
        }

        private function onReadComplete(e:Event):void {
            var loader:Loader = new Loader();
            loader.contentLoaderInfo.addEventListener(Event.COMPLETE, onDecodeComplete);
            loader.loadBytes(mFileByteArray);
        }

        private function onIOError(e:Event):void {
            mReadError = true;
        }
        
        public function setInitFile(fn:String):void {
            if (mFileName != fn) mIsLoaded = false;
            mFileName = fn;
        }
        
        public function get texture():Texture {
            return mTexture;
        }
        
        public function addEventListener(event:int, fun:Function):void {
            if (event < MyResource.LOAD_COMPLETE || event > MyResource.LOAD_COMPLETE) return;
            mEventListeners[event].push(fun);
        }
        
        public function addUser():void {
            mUserCount++;
        }
        
        public function getUserCount():int {
            return mUserCount;
        }
        
        public function load():void {
            if (null == mFileName || 0 == mFileName.length) return;
            var file:URLStream = new URLStream();
            file.addEventListener(IOErrorEvent.IO_ERROR, onIOError);
            file.addEventListener(ProgressEvent.PROGRESS, onReading);
            file.addEventListener(Event.COMPLETE, onReadComplete);
            mIsLoaded = false;
            mReadError = false;
            mFileByteArray.length = 0;
            mReadingPos = 0;
            file.load(new URLRequest(mFileName));
        }
        
        public function isLoaded():Boolean {
            return mIsLoaded;
        }
        
        public function removeEventListener(event:int, fun:Function):void {
            if (event < MyResource.LOAD_COMPLETE || event > MyResource.LOAD_COMPLETE) return;
            var pos:int = mEventListeners[event].indexOf(fun);
            if (pos < 0) return;
            mEventListeners[event].splice(pos, 1);
        }
        
        public function removeUser():void {
            mUserCount--;
        }
        
        public function unload():void {
            mIsLoaded = false;
        }
        
    }
}
