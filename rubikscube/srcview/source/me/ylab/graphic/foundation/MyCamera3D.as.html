package me.ylab.graphic.foundation {
    import com.adobe.utils.PerspectiveMatrix3D;
    
    import me.ylab.graphic.types.GraphicMatrix4x4;
    import me.ylab.tools.MyCoordinateSystem3;
    import me.ylab.tools.MyMatrix4x4;
    import me.ylab.tools.MyRay3;
    import me.ylab.tools.MySize2;
    import me.ylab.tools.MyVector3;
    import me.ylab.tools.MyVector4;
    import me.ylab.tools.MyVertex2;
    import me.ylab.tools.MyVertex3;
    import me.ylab.tools.MyVertex4;

    public class MyCamera3D {

        public static const IN_WORLD_SPACE:int = 0;
        /**
        * in camera space
        */
        public static const IN_CAMERA_SPACE:int = 1;
        
        public static const FIXED_EYE:int = 0;
        public static const FIXED_EYE_ON:int = 1;

        protected var mProjMatrix:GraphicMatrix4x4;
        protected var mCombineMatrix:GraphicMatrix4x4;
        protected var mProjFaceDistance:Number;
        protected var mViewPortLT:MyVertex2;
        protected var mViewPortRB:MyVertex2;
        protected var mViewPortSize:MySize2;
        protected var mProjFaceLT:MyVertex2;
        protected var mProjFaceRB:MyVertex2;
        protected var mProjFaceSize:MySize2;
        protected var mCoordinate:MyCoordinateSystem3;
        
        protected var mEye:MyVertex3;
        protected var mEyeOn:MyVertex3; // in camera coordinate
        protected var mEyeUp:MyVector3;
        protected var mTmpSTW:MyMatrix4x4;
        protected var mTmpWTS:MyMatrix4x4;
        protected var mTmpVector4:MyVertex4;
        
        private var tRay:MyRay3;
        private var tMousePointInWorld:MyVector4;

        public function MyCamera3D() {
            mProjMatrix = new GraphicMatrix4x4;
            mCombineMatrix = new GraphicMatrix4x4;
            mViewPortLT = new MyVertex2;
            mViewPortRB = new MyVertex2;
            mViewPortSize = new MySize2;
            mProjFaceLT = new MyVertex2;
            mProjFaceRB = new MyVertex2;
            mProjFaceSize = new MySize2;
            mCoordinate = new MyCoordinateSystem3;
            mEye = new MyVertex3;
            mEyeOn = new MyVertex3;
            mEyeUp = new MyVector3;
            
            mTmpSTW = new MyMatrix4x4;
            mTmpWTS = new MyMatrix4x4;
            mTmpVector4 = new MyVertex4;
            
            tRay = new MyRay3();
            tMousePointInWorld = new MyVector4();
            
            mEye.setXYZ(0, 0, 0);
            mEyeOn.setXYZ(0, 0, -1);
            mEyeUp.setXYZ(0, 1, 0);
        }
        
        public function getEyeRayByScreenPoint(x:int, y:int):MyRay3 {
            tRay.start.setXYZ(mEye.x, mEye.y, mEye.z);
            
            tMousePointInWorld.setXYZW(0, 0, -mProjFaceDistance, 1);
            tMousePointInWorld.x = mProjFaceLT.x + (x - mViewPortLT.x) * mProjFaceSize.width / mViewPortSize.width;
            tMousePointInWorld.y = mProjFaceRB.y + (y - mViewPortRB.y) * mProjFaceSize.height / mViewPortSize.height;
            
            mCoordinate.selfToWorld.multiplyToVertex(tMousePointInWorld);
            tMousePointInWorld.setXYZ(tMousePointInWorld.x - mEye.x, tMousePointInWorld.y - mEye.y, tMousePointInWorld.z - mEye.z);
            tMousePointInWorld.normalize();
            tRay.direction.setXYZ(tMousePointInWorld.x, tMousePointInWorld.y, tMousePointInWorld.z);

            return tRay;
        }
        
        public function worldToScreen(world:MyVertex4, screen:MyVertex2):void {
            var tScreen:MyVertex4 = this.matrix.multiplyVertex(world);
            tScreen.setXY(tScreen.x / tScreen.w, tScreen.y / tScreen.w);
            screen.x = mViewPortLT.x + (tScreen.x + 1) * mViewPortSize.width / 2;
            screen.y = mViewPortRB.y + (tScreen.y + 1) * mViewPortSize.height / 2;
        }
        
        public function setViewPort(left:Number, top:Number, right:Number, bottom:Number):void {
            mViewPortLT.setXY(left, top);
            mViewPortRB.setXY(right, bottom);
            mViewPortSize.setWidthHeight(right - left + 1, top - bottom - 1);
        }
        
        public function get viewPortSize():MySize2 {
            return new MySize2(mViewPortSize.width, mViewPortSize.height);
        }
        
        public function setEyeParam(eye:MyVertex3, on:MyVertex3, up:MyVector3):void {
            var eyeTo:MyVector3 = new MyVector3(eye.x - on.x, eye.y - on.y, eye.z - on.z);
            mEyeOn.setXYZ(0, 0, -eyeTo.module);
            eyeTo.normalize();
            var right:MyVector3 = MyVector3.CrossProduct(up, eyeTo);
            right.normalize();
            var eyeUp:MyVector3 = MyVector3.CrossProduct(eyeTo, right);
            eyeUp.normalize();
            var eyeTrans:MyVector3 = new MyVector3(eye.x, eye.y, eye.z);
            
            mEye.setXYZ(eye.x, eye.y, eye.z);
            mEyeUp.setXYZ(eyeUp.x, eyeUp.y, eyeUp.z);
            
            mCoordinate.returnToOrigin();
            
            mCoordinate.worldToSelf.m[0] = right.x;
            mCoordinate.worldToSelf.m[1] = eyeUp.x;
            mCoordinate.worldToSelf.m[2] = eyeTo.x;
            
            mCoordinate.worldToSelf.m[4] = right.y;
            mCoordinate.worldToSelf.m[5] = eyeUp.y;
            mCoordinate.worldToSelf.m[6] = eyeTo.y;
            
            mCoordinate.worldToSelf.m[8] = right.z;
            mCoordinate.worldToSelf.m[9] = eyeUp.z;
            mCoordinate.worldToSelf.m[10] = eyeTo.z;
            
            mCoordinate.worldToSelf.m[12] = -MyVector3.DotProduct(right, eyeTrans);
            mCoordinate.worldToSelf.m[13] = -MyVector3.DotProduct(eyeUp, eyeTrans);
            mCoordinate.worldToSelf.m[14] = -MyVector3.DotProduct(eyeTo, eyeTrans);
            
            mCoordinate.selfToWorld.m[0] = right.x;
            mCoordinate.selfToWorld.m[1] = right.y;
            mCoordinate.selfToWorld.m[2] = right.z;
            
            mCoordinate.selfToWorld.m[4] = eyeUp.x;
            mCoordinate.selfToWorld.m[5] = eyeUp.y;
            mCoordinate.selfToWorld.m[6] = eyeUp.z;
            
            mCoordinate.selfToWorld.m[8] = eyeTo.x;
            mCoordinate.selfToWorld.m[9] = eyeTo.y;
            mCoordinate.selfToWorld.m[10] = eyeTo.z;
            
            mCoordinate.selfToWorld.m[12] = eyeTrans.x;
            mCoordinate.selfToWorld.m[13] = eyeTrans.y;
            mCoordinate.selfToWorld.m[14] = eyeTrans.z;
            
        }
        
        public function get eye():MyVertex3 {
            return new MyVertex3(mEye.x, mEye.y, mEye.z);
        }
        
        public function get eyeOn():MyVertex3 {
            return new MyVertex3(mEyeOn.x, mEyeOn.y, mEyeOn.z);
        }
        
        public function get eyeUp():MyVector3 {
            return new MyVector3(mEyeUp.x, mEyeUp.y, mEyeUp.z);
        }
        
        public function set eye(v:MyVertex3):void {
            setEyeParam(v, mEyeOn, mEyeUp);
        }
        
        public function set eyeOn(v:MyVertex3):void {
            setEyeParam(mEye, v, mEyeUp);
        }
        
        public function set eyeUp(v:MyVector3):void {
            setEyeParam(mEye, mEyeOn, v);
        }
        
        public function get matrix():GraphicMatrix4x4 {
            mCombineMatrix.copyFrom(mProjMatrix);
            mCombineMatrix.multiply(mCoordinate.worldToSelf);
            return mCombineMatrix;
        }
        
        public function get viewMatrix():GraphicMatrix4x4 {
            mCombineMatrix.copyFrom(mCoordinate.worldToSelf);
            return mCombineMatrix;
        }
        
        public function reset():void {
            mCoordinate.returnToOrigin();
            mEye.setXYZ(0, 0, 0);
            mEyeOn.setXYZ(0, 0, -1);
            mEyeUp.setXYZ(0, 1, 0);
        }
        
        public function resetRotation():void {
            var mX:Number = mCoordinate.selfToWorld.m[12];
            var mY:Number = mCoordinate.selfToWorld.m[13];
            var mZ:Number = mCoordinate.selfToWorld.m[14];
            mCoordinate.returnToOrigin();
            mCoordinate.selfToWorld.m[12] = mX;
            mCoordinate.selfToWorld.m[13] = mY;
            mCoordinate.selfToWorld.m[14] = mZ;
            mCoordinate.worldToSelf.m[12] = -mX;
            mCoordinate.worldToSelf.m[13] = -mY;
            mCoordinate.worldToSelf.m[14] = -mZ;
        }

        public function rotateModelToFace(model:MyModel):void {
            model.coordinate.returnToOrigin();
            model.coordinate.selfToWorld.resetBy(mCoordinate.selfToWorld);
            model.coordinate.worldToSelf.resetBy(mCoordinate.worldToSelf);
            model.coordinate.selfToWorld.m[12] = 0;
            model.coordinate.selfToWorld.m[13] = 0;
            model.coordinate.selfToWorld.m[14] = 0;
            model.coordinate.worldToSelf.m[12] = 0;
            model.coordinate.worldToSelf.m[13] = 0;
            model.coordinate.worldToSelf.m[14] = 0;
        }
        
        public function syncModelCoordinate(model:MyModel):void {
            model.coordinate.returnToOrigin();
            model.coordinate.selfToWorld.resetBy(mCoordinate.selfToWorld);
            model.coordinate.worldToSelf.resetBy(mCoordinate.worldToSelf);
        }
        
        public function moveAll(x:Number, y:Number, z:Number, space:int = 0):void {
            if (IN_WORLD_SPACE == space) {
                mCoordinate.insertTranslate(x, y, z);
            } else if (IN_CAMERA_SPACE == space) {
                
                var stw:MyMatrix4x4 = MyMatrix4x4.CreateFrom(mCoordinate.selfToWorld);
                var wts:MyMatrix4x4 = MyMatrix4x4.CreateFrom(mCoordinate.worldToSelf);
                
                mCoordinate.returnToOrigin();
                mCoordinate.insertTranslate(x, y, z);
                
                mCoordinate.worldToSelf.multiply(wts);
                mCoordinate.selfToWorld.resetBy(stw.multiply(mCoordinate.selfToWorld));
            }
            mEye.setXYZ(mCoordinate.selfToWorld.m[12], mCoordinate.selfToWorld.m[13], mCoordinate.selfToWorld.m[14]);
        }
        
        public function moveEye(x:Number, y:Number, z:Number, space:int = 0):void {
            var mEyeOnVector4:MyVertex4 = new MyVertex4;
            mEyeOnVector4.setXYZ(0, 0, mEyeOn.z);
            mEyeOnVector4 = mCoordinate.selfToWorld.multiplyVertex(mEyeOnVector4);
            if (IN_WORLD_SPACE == space) {
                mEye.setXYZ(mEye.x + x, mEye.y + y, mEye.z + z);
            } else if (IN_CAMERA_SPACE == space) {
                var mEyeVector4:MyVertex4 = new MyVertex4;
                mEyeVector4.setXYZ(x, y, z);
                mEyeVector4 = mCoordinate.selfToWorld.multiplyVertex(mEyeVector4);
                mEye.setXYZ(mEyeVector4.x, mEyeVector4.y, mEyeVector4.z);
            }
            setEyeParam(mEye, mEyeOnVector4, mEyeUp);
        }
        
        public function rotate(x:Number, y:Number, z:Number, style:int = 0):void {
            mTmpSTW.resetBy(mCoordinate.selfToWorld);
            mTmpWTS.resetBy(mCoordinate.worldToSelf);
            
            if (FIXED_EYE == style) {
                mCoordinate.returnToOrigin();
                mCoordinate.insertRotateByX(x);
                mCoordinate.insertRotateByY(y);
                mCoordinate.insertRotateByZ(z);
                
                mCoordinate.worldToSelf.multiply(mTmpWTS);
                mCoordinate.selfToWorld.resetBy(mTmpSTW.multiply(mCoordinate.selfToWorld));
                
            } else if (FIXED_EYE_ON == style) {
                
                mCoordinate.returnToOrigin();
                mCoordinate.insertTranslate(0, 0, -mEyeOn.z);
                mCoordinate.insertRotateByX(x);
                mCoordinate.insertRotateByY(y);
                mCoordinate.insertRotateByZ(z);
                mCoordinate.insertTranslate(0, 0, mEyeOn.z);
                
                mCoordinate.worldToSelf.multiply(mTmpWTS);
                mCoordinate.selfToWorld.resetBy(mTmpSTW.multiply(mCoordinate.selfToWorld));
                
                mEye.setXYZ(mCoordinate.selfToWorld.m[12], mCoordinate.selfToWorld.m[13], mCoordinate.selfToWorld.m[14]);
            }
            mTmpVector4.setXYZ(0, 1, 0);
            mCoordinate.selfToWorld.multiplyToVertex(mTmpVector4);
            mEyeUp.setXYZ(mTmpVector4.x - mEye.x, mTmpVector4.y - mEye.y, mTmpVector4.z - mEye.z);
            mEyeUp.normalize();
        }
        
        // fov unit is radian, a circle is 2*pi.
        public function setPerspectiveByFovY(fov:Number, ratioXY:Number, zNear:Number = 0.1, zFar:Number = 1000):void {

            mProjFaceDistance = zNear;
            
            var top:Number = Math.tan(fov/2) * zNear;
            var bottom:Number = -top;
            var right:Number = top * ratioXY;
            var left:Number = -right;
            
            mProjFaceLT.setXY(left, top);
            mProjFaceRB.setXY(right, bottom);
            mProjFaceSize.setWidthHeight(right - left, top - bottom);
            
            mProjMatrix.identify();
            
            mProjMatrix.m[0] = (zNear + zNear) / (right - left);
            mProjMatrix.m[5] = (zNear + zNear) / (top - bottom);

            mProjMatrix.m[8] = (right + left) / (right - left);
            mProjMatrix.m[9] = (top + bottom) / (top - bottom);
            mProjMatrix.m[10] = -(zFar + zNear) / (zFar - zNear);
            mProjMatrix.m[11] = -1;
            
            mProjMatrix.m[14] = -2 * zFar * zNear / (zFar - zNear);
            mProjMatrix.m[15] = 0;
            
        }
    }
}
