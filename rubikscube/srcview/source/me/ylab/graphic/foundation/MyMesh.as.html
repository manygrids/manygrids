package me.ylab.graphic.foundation {
    
    import me.ylab.tools.MyCoordinateSystem3;
    import me.ylab.tools.MyMatrix4x4;
    import me.ylab.tools.MyRay3;
    import me.ylab.tools.MyTriangle;
    import me.ylab.tools.MyVector3;
    import me.ylab.tools.MyVector4;
    import me.ylab.tools.MyVertex3;
    import me.ylab.tools.MyVertex4;

    public class MyMesh {

        protected var mGraphic:Graphic3D;
        protected var mTriangles:Vector.<MyTriangle>;
        protected var mCoordinate:MyCoordinateSystem3;
        protected var mModel:MyModel;
        protected var mPrepared:Boolean;
        protected var mColor:MyColor4f;
        
        protected var mIsTransparent:Boolean;

        private var tStart:MyVector4;
        private var tDir:MyVector4;
        private var tRay:MyRay3;
        private var tCrossTestResult:CrossTestResult;
        
        public function MyMesh(graphic:Graphic3D = null,
                               color:MyColor4f = null) {
            mTriangles = new Vector.<MyTriangle>;
            mCoordinate = new MyCoordinateSystem3;
            mPrepared = false;
            mGraphic = graphic;
            mColor = color;
            if (null == mColor) mColor = new MyColor4f(1, 1, 1, 1);

            tStart = new MyVector4;
            tDir = new MyVector4;
            tRay = new MyRay3;
            tCrossTestResult = new CrossTestResult();
        }

        public function get normal():MyVector4 {
            return mTriangles[0].normal;
        }
        
        public function setColor(r:Number, g:Number, b:Number, a:Number):void {
            mColor.setRGBA(r, g, b, a);
        }
        
        public function get color():MyColor4f {
            return new MyColor4f(mColor.r, mColor.g, mColor.b, mColor.a);
        }
        
        public function set graphic(g:Graphic3D):void {
            mGraphic = g;
        }
        
        public function get graphic():Graphic3D {
            return mGraphic;
        }

        public function addTo(model:MyModel):void {
            mModel = model;
        }
        
        public function removeFrom(model:MyModel):void {
            mModel = null;
        }
        
        public function get model():MyModel {
            return mModel;
        }
        
        public function get isTransparent():Boolean {
            return mIsTransparent;
        }
        
        public function set transparent(trans:Boolean):void {
            if (mPrepared) {
                unprepare();
                mIsTransparent = trans;
                prepareDraw();
            } else {
                mIsTransparent = trans;
            }
        }
        
        public function addTriangle(tri:MyTriangle):void {
            mTriangles.push(tri);
        }
        
        public function get coordinate():MyCoordinateSystem3 {
            return mCoordinate;
        }
        
        public function set coordinate(coord:MyCoordinateSystem3):void {
            mCoordinate = coord;
        }
        
        public function prepareDraw():void {
            if (mPrepared) return;
            mPrepared = true;
        }
        
        public function unprepare():void {
            if (!mPrepared) return;
            mPrepared = false;
        }
        
        public function draw():void {}
        
        public function fillDrawMatrix(matrix:MyMatrix4x4):void {
            mModel.fillDrawMatrix(matrix);
            matrix.multiply(mCoordinate.selfToWorld);
        }

        public function obtainWorldCoordinate():MyCoordinateSystem3 {
            var father:MyCoordinateSystem3 = mModel.obtainWorldCoordinate();
            father.selfToWorld.multiply(mCoordinate.selfToWorld);
            father.worldToSelf.resetBy(MyMatrix4x4.Multiply(mCoordinate.worldToSelf, father.worldToSelf));
            return father;
        }
        
        public function testCrossByRay(ray:MyRay3):CrossTestResult {
            var dis:Number = Number.MAX_VALUE;
            var tDis:Number = 0;
            var crossVertex:MyVertex4 = null;
            tCrossTestResult.result = false;
            
            tStart.setXYZW(ray.start.x, ray.start.y, ray.start.z, 1);
            tDir.setXYZW(ray.start.x + ray.direction.x, ray.start.y + ray.direction.y, ray.start.z + ray.direction.z, 1);
            mCoordinate.worldToSelf.multiplyToVertex(tStart);
            mCoordinate.worldToSelf.multiplyToVertex(tDir);
            tRay.start.setXYZ(tStart.x, tStart.y, tStart.z);
            tRay.direction.resetBy4(tDir.minus(tStart));
            
            for (var i:int=0; i<mTriangles.length; i++) {
                crossVertex = mTriangles[i].getCrossVertex(tRay);
                if (null != crossVertex) {
                    tDis =  (crossVertex.x - tRay.start.x) * (crossVertex.x - tRay.start.x) + 
                            (crossVertex.y - tRay.start.y) * (crossVertex.y - tRay.start.y) +
                            (crossVertex.z - tRay.start.z) * (crossVertex.z - tRay.start.z);
                        
                    if (tDis < dis) {
                        dis = tDis;
                        tCrossTestResult.result = true;
                        tCrossTestResult.tri = mTriangles[i];
                        tCrossTestResult.vertex = crossVertex;
                    }
                }
            }
            
            if (tCrossTestResult.result) {
                mCoordinate.selfToWorld.multiplyVertexAtoB(tCrossTestResult.vertex, tCrossTestResult.vertex);
                tCrossTestResult.mesh = this;
            }
            
            return tCrossTestResult;
        }
    }
}
