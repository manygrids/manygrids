package me.ylab.graphic.foundation {

    public class MyColor4f {

        public static const black:MyColor4f = new MyColor4f(0, 0, 0, 1);
        public static const white:MyColor4f = new MyColor4f(1, 1, 1, 1);
        public static const gray:MyColor4f  = new MyColor4f(0.5, 0.5, 0.5, 1);
        public static const red:MyColor4f   = new MyColor4f(1, 0, 0, 1);
        public static const green:MyColor4f = new MyColor4f(0, 1, 0, 1);
        public static const blue:MyColor4f  = new MyColor4f(0, 0, 1, 1);

        protected var mRed:Number;
        protected var mGreen:Number;
        protected var mBlue:Number;
        protected var mAlpha:Number;

        public function MyColor4f(r:Number = 0, g:Number = 0, b:Number = 0, a:Number = 0) {
            mRed = r;
            mGreen = g;
            mBlue = b;
            mAlpha = a;
        }
        
        public function setRGBA(r:Number, g:Number, b:Number, a:Number):void {
            mRed = r;
            mGreen = g;
            mBlue = b;
            mAlpha = a;
        }
        
        public function get argb():int {
            var ret:int = 0;
            ret = (int(a * 255)) << 24 | (int(r * 255)) << 16 | (int(g * 255)) << 8 | (int(b * 255));
            return ret;
        }
        
        public function get rgb():int {
            var ret:int = 0;
            ret = (int(r * 255)) << 16 | (int(g * 255)) << 8 | (int(b * 255));
            return ret;
        }
        
        public function get r():Number {
            return mRed;
        }
        
        public function get g():Number {
            return mGreen;
        }
        
        public function get b():Number {
            return mBlue;
        }
        
        public function get a():Number {
            return mAlpha;
        }

        public function set r(v:Number):void {
            mRed = r;
        }

        public function set g(v:Number):void {
            mGreen = g;
        }

        public function set b(v:Number):void {
            mBlue = b;
        }

        public function set a(v:Number):void {
            mAlpha = a;
        }
        
        public function add(c:MyColor4f):MyColor4f {
            mRed += c.r;
            mGreen += c.g;
            mBlue += c.b;
            mAlpha += c.a;
            return this;
        }

        public function minus(c:MyColor4f):MyColor4f {
            mRed -= c.r;
            mGreen -= c.g;
            mBlue -= c.b;
            mAlpha -= c.a;
            return this;
        }
    }
}
