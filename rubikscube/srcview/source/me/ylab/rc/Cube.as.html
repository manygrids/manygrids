package me.ylab.rc {

    import me.ylab.graphic.foundation.MyModel;
    import me.ylab.tools.*;
    
    public class Cube {

        private var mPos:MyVector3 = new MyVector3();
        private var mRotate:MyVector3 = new MyVector3();
        private var mRotateMatrix:MyMatrix4x4= new MyMatrix4x4();
        private var mXDir:MyVector4 = new MyVector4();
        private var mYDir:MyVector4 = new MyVector4();
        
        private var mIndex:int;
        private var mModel:MyModel;
        
        public function Cube(index:int, model:MyModel) {
            mIndex = index;
            mModel = model;
            mXDir.setXYZ(1, 0, 0);
            mYDir.setXYZ(0, 1, 0);
        }
        
        public function get model():MyModel {
            return mModel;
        }
        
        public function set model(m:MyModel):void {
            mModel = m;
        }

        public function resetRotate():void {
            mRotate.setXYZ(0, 0, 0);
            mXDir.setXYZ(1, 0, 0);
            mYDir.setXYZ(0, 1, 0);
        }
        
        public function refreshRotate():void {
            var matrix:MyMatrix4x4 = null;
            matrix = MyMatrix4x4.GetRotateMatrix(mModel.coordinate.selfToWorld);
            mXDir.setXYZ(1, 0, 0);
            mYDir.setXYZ(0, 1, 0);
            matrix.multiplyToVertex(mXDir);
            matrix.multiplyToVertex(mYDir);
        }
        
        public function get index():int {
            return mIndex;
        }
        
        public function get pos():MyVector3 {
            return new MyVector3(mPos.x, mPos.y, mPos.z);
        }
        
        public function get rotateMatrix():MyMatrix4x4 {
            return MyMatrix4x4.CreateFrom(mRotateMatrix);
        }

        public function get xDirection():MyVector4 {
            return new MyVector4(mXDir.x, mXDir.y, mXDir.z);
        }

        public function get yDirection():MyVector4 {
            return new MyVector4(mYDir.x, mYDir.y, mYDir.z);
        }

        public function toString():String {
            return "(" + ((mIndex<10)?"0":"") + mIndex.toString() + mXDir.toString() + mYDir.toString() + ")";
        }
        
        public function loadFromString(str:String):void {
            var op:int = 0;
            var dir:int = 0;
            var ang:int = 0;
            var zeroCode:int = '0'.charCodeAt(0);
            for (var i:int=0; i<str.length; i++) {
                op = str.charCodeAt(i) - zeroCode - 1;
                dir = op / 3;
                ang = op % 3 + 1;
                switch (dir) {
                case 0:
                    mModel.coordinate.addTranslate(0.5, 0.5, 0.5);
                    mModel.coordinate.addRotateByX(ang * 90);
                    mModel.coordinate.addTranslate(-0.5, -0.5, -0.5);
                    break;
                case 1:
                    mModel.coordinate.addTranslate(0.5, 0.5, 0.5);
                    mModel.coordinate.addRotateByY(ang * 90);
                    mModel.coordinate.addTranslate(-0.5, -0.5, -0.5);
                    break;
                case 2:
                    mModel.coordinate.addTranslate(0.5, 0.5, 0.5);
                    mModel.coordinate.addRotateByZ(ang * 90);
                    mModel.coordinate.addTranslate(-0.5, -0.5, -0.5);
                    break;
                }
            }

            refreshRotate();
        }
        
        public function saveToString():String {
            var tmpRet:String = "";
            
            var matrix:MyMatrix4x4 = null;
            
            var tXDir:MyVector4 = new MyVector4(mXDir.x, mXDir.y, mXDir.z, mXDir.w);
            var tYDir:MyVector4 = new MyVector4(mYDir.x, mYDir.y, mYDir.z, mYDir.w);
            
            // rotate by model y, make tXDir to xz plan
            if (Math.abs(tXDir.dotProduct(new MyVector4(0, 0, 1))) > 0.9
                || Math.abs(tXDir.dotProduct(new MyVector4(1, 0, 0))) > 0.9) {
                // xDir is on x or z axis
                tmpRet += "y0";
            } else if (tXDir.dotProduct(new MyVector4(0, 1, 0)) > 0.9) {
                // xDir is on y positive axis
                if (Math.abs(tYDir.dotProduct(new MyVector4(1, 0, 0))) > 0.9) {
                    // yDir is on x axis
                    matrix = MyMatrix4x4.getRotateByXMatrix(90);
                    if (tYDir.dotProduct(new MyVector4(1, 0, 0)) > 0.9) {
                        // yDir is on x positive axis
                        tmpRet += "y1";
                    } else {
                        // yDir is on x negative axis
                        tmpRet += "y3";
                    }
                } else {
                    // yDir is on z axis
                    matrix = MyMatrix4x4.getRotateByZMatrix(-90);
                    if (tYDir.dotProduct(new MyVector4(0, 0, 1)) > 0.9) {
                        // yDir is on z positive axis
                        tmpRet += "y3";
                    } else {
                        // yDir is on z negative axis
                        tmpRet += "y1";
                    }
                }
                
                matrix.multiplyToVertex(tXDir);
                matrix.multiplyToVertex(tYDir);
            } else {
                // xDir is on y negative axis
                if (Math.abs(tYDir.dotProduct(new MyVector4(1, 0, 0))) > 0.9) {
                    // yDir is on x axis
                    matrix = MyMatrix4x4.getRotateByXMatrix(-90);
                    if (tYDir.dotProduct(new MyVector4(1, 0, 0)) > 0.9) {
                        // yDir is on x positive axis
                        tmpRet += "y3";
                    } else {
                        // yDir is on x negative axis
                        tmpRet += "y1";
                    }
                } else {
                    // yDir is on z axis
                    matrix = MyMatrix4x4.getRotateByZMatrix(90);
                    if (tYDir.dotProduct(new MyVector4(0, 0, 1)) > 0.9) {
                        // yDir is on z positive axis
                        tmpRet += "y1";
                    } else {
                        // yDir is on z negative axis
                        tmpRet += "y3";
                    }
                }
                
                matrix.multiplyToVertex(tXDir);
                matrix.multiplyToVertex(tYDir);
            }
            
            // rotate by y or z, make tXDir to x positive axis
            if (Math.abs(tXDir.dotProduct(new MyVector4(0, 0, 1))) > 0.9) {
                // tXDir is on z axis
                if (tXDir.dotProduct(new MyVector4(0, 0, 1)) > 0.9) {
                    // tXDir is on z positive axis
                    matrix = MyMatrix4x4.getRotateByYMatrix(90);
                    if (Math.abs(tYDir.dotProduct(new MyVector4(0, 1, 0))) > 0.9) {
                        // tYDir is on y axis
                        if (tYDir.dotProduct(new MyVector4(0, 1, 0)) > 0.9) {
                            // tYDir is on y positive axis
                            tmpRet += "y1";
                        } else {
                            // tYDir is on y negative axis
                            tmpRet += "y3";
                        }
                    } else {
                        // tYDir is on x axis
                        if (tYDir.dotProduct(new MyVector4(1, 0, 0)) > 0.9) {
                            // tYDir is on x positive axis
                            tmpRet += "z1";
                        } else {
                            // tYDir is on x negative axis
                            tmpRet += "z3";
                        }
                    }
                    matrix.multiplyToVertex(tXDir);
                    matrix.multiplyToVertex(tYDir);
                } else {                    
                    // tXDir is on z negative axis
                    matrix = MyMatrix4x4.getRotateByYMatrix(-90);
                    if (Math.abs(tYDir.dotProduct(new MyVector4(0, 1, 0))) > 0.9) {
                        // tYDir is on y axis
                        if (tYDir.dotProduct(new MyVector4(0, 1, 0)) > 0.9) {
                            // tYDir is on y positive axis
                            tmpRet += "y3";
                        } else {
                            // tYDir is on y negative axis
                            tmpRet += "y1";
                        }
                    } else {
                        // tYDir is on x axis
                        if (tYDir.dotProduct(new MyVector4(1, 0, 0)) > 0.9) {
                            // tYDir is on x positive axis
                            tmpRet += "z1";
                        } else {
                            // tYDir is on x negative axis
                            tmpRet += "z3";
                        }
                    }
                    matrix.multiplyToVertex(tXDir);
                    matrix.multiplyToVertex(tYDir);
                }
            } else if (tXDir.dotProduct(new MyVector4(1, 0, 0)) > 0.9) {
                // tXDir is on x positive axis
                tmpRet += "y0";
            } else {
                // tXDir is on x negative axis
                if (Math.abs(tYDir.dotProduct(new MyVector4(0, 1, 0))) > 0.9) {
                    // tYDir is on y axis
                    matrix = MyMatrix4x4.getRotateByYMatrix(180);
                } else {
                    // tYDir is on z axis
                    matrix = MyMatrix4x4.getRotateByZMatrix(180);
                }
                tmpRet += "y2";
                matrix.multiplyToVertex(tXDir);
                matrix.multiplyToVertex(tYDir);
            }

            if (tXDir.dotProduct(new MyVector4(1, 0, 0)) > 0.9) {
                
            } else {
                trace("error");
            }
            
            // rotate by x, make tYDir to y positive axis
            if (Math.abs(tYDir.dotProduct(new MyVector4(0, 0, 1))) > 0.9) {
                // tYDir is on z axis
                if (tYDir.dotProduct(new MyVector4(0, 0, 1)) > 0.9) {
                    // tYDir is on z positive axis
                    matrix = MyMatrix4x4.getRotateByXMatrix(-90);
                    tmpRet += "x3";
                } else {                    
                    // tYDir is on z negative axis
                    matrix = MyMatrix4x4.getRotateByXMatrix(90);
                    tmpRet += "x1";
                }
                matrix.multiplyToVertex(tXDir);
                matrix.multiplyToVertex(tYDir);
            } else if (tYDir.dotProduct(new MyVector4(0, 1, 0)) > 0.9) {
                // tYDir is on y positive axis
                tmpRet += "x0";
            } else {
                // mYDir is on y negative axis
                matrix = MyMatrix4x4.getRotateByXMatrix(180);
                matrix.multiplyToVertex(tXDir);
                matrix.multiplyToVertex(tYDir);
                tmpRet += "x2";
            }
            
            var xCode:int = 'x'.charCodeAt(0);
            var ret:String = '';
            for (var i:int=0; i<tmpRet.length; i++) {
                if ('0' == tmpRet.charAt(i+1)) {
                } else if ('1' == tmpRet.charAt(i+1)) {
                    ret = ((tmpRet.charAt(i).charCodeAt(0) - xCode) * 3 + 3) + ret;
                } else if ('3' == tmpRet.charAt(i+1)) {
                    ret = ((tmpRet.charAt(i).charCodeAt(0) - xCode) * 3 + 1) + ret;
                } else {
                    ret = ((tmpRet.charAt(i).charCodeAt(0) - xCode) * 3 + 2) + ret;
                }
                i++;
            }
            
            return ret;
        }
    }
