package me.ylab.rc.animations {

    import me.ylab.application.MyAnimation;
    import me.ylab.rc.Cube;
    import me.ylab.rc.CubeGroup;
    
    public class CubeOperateAnimation extends MyAnimation {

        protected var mCubeGroup:CubeGroup;
        protected var mRotateCubes:Vector.<Cube>;
        protected var mRotateDir:int;
        protected var mCurrentAngle:Number;
        protected var mEndAngle:Number;
        protected var mSpeed:Number;
        
        public function CubeOperateAnimation(cubeGroup:CubeGroup,
                                             rotateCubes:Vector.<Cube>,
                                             rotateDir:int,
                                             startAngle:Number,
                                             endAngle:Number,
                                             speed:Number,
                                             inter:int = 50) {
            super();
            mCubeGroup = cubeGroup;
            mRotateCubes = rotateCubes;
            mRotateDir = rotateDir;
            mCurrentAngle = startAngle;
            mEndAngle = endAngle;
            mSpeed = speed;
            mInterval = inter;
        }
        
        public override function animate(lapseTime:int):void {
            if (mCurrentAngle == mEndAngle) {
                for each (var f:Function in mEventListeners[ON_ANIMATION_END]) {
                    f.call(null, this);
                }
                return;
            }
            var rotateAngle:Number = mSpeed * (lapseTime / mInterval);
            mCurrentAngle += rotateAngle;
            if (mSpeed > 0) {
                if (mCurrentAngle >= mEndAngle) {
                    rotateAngle = mEndAngle - (mCurrentAngle - rotateAngle);
                    mCurrentAngle = mEndAngle;
                }
            } else if (mSpeed < 0) {
                if (mCurrentAngle <= mEndAngle) {
                    rotateAngle = mEndAngle - (mCurrentAngle - rotateAngle);
                    mCurrentAngle = mEndAngle;
                }
            }
            mCubeGroup.helperModel.coordinate.returnToOrigin();
            switch (mRotateDir) {
                case CubeGroup.AROUND_X:
                    mCubeGroup.helperModel.coordinate.insertRotateByX(mCurrentAngle);
                    break;
                case CubeGroup.AROUND_Y:
                    mCubeGroup.helperModel.coordinate.insertRotateByY(mCurrentAngle);
                    break;
                case CubeGroup.AROUND_Z:
                    mCubeGroup.helperModel.coordinate.insertRotateByZ(mCurrentAngle);
                    break;
            }
        }
    }
}
