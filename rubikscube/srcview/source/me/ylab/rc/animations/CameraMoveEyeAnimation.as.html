package me.ylab.rc.animations {
    
    import me.ylab.application.MyAnimation;
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.foundation.MyCamera3D;
    import me.ylab.rc.CubeGroup;
    
    public class CameraMoveEyeAnimation extends MyAnimation {
        
        protected var mGraphic:Graphic3D;
        protected var mBaseDir:int;
        protected var mCurrentPos:Number;
        protected var mEndPos:Number;
        protected var mSpeed:Number;
        
        public function CameraMoveEyeAnimation(graphic:Graphic3D,
                                               baseDir:int,
                                               endPos:Number,
                                               speed:Number,
                                               inter:int = 50) {
            super();
            mGraphic = graphic;
            mBaseDir = baseDir;
            mCurrentPos = 0;
            mEndPos = endPos;
            mSpeed = speed;
            mInterval = inter;
        }
        
        public override function animate(lapseTime:int):void {
            if (mCurrentPos == mEndPos) {
                for each (var f:Function in mEventListeners[ON_ANIMATION_END]) {
                    f.call(null, this);
                }
                if (mLoop) {
                    mCurrentPos = 0;
                } else {
                    return;
                }
            }
            var moveDistance:Number = mSpeed * (lapseTime / mInterval);
            mCurrentPos += moveDistance;
            if (mSpeed > 0) {
                if (mCurrentPos >= mEndPos) {
                    moveDistance = mEndPos - (mCurrentPos - moveDistance);
                    mCurrentPos = mEndPos;
                }
            } else if (mSpeed < 0) {
                if (mCurrentPos <= mEndPos) {
                    moveDistance = mEndPos - (mCurrentPos - moveDistance);
                    mCurrentPos = mEndPos;
                }
            }
            switch (mBaseDir) {
                case CubeGroup.AROUND_X:
                    mGraphic.currentCamera.moveEye(moveDistance, 0, 0, MyCamera3D.IN_CAMERA_SPACE);
                    break;
                case CubeGroup.AROUND_Y:
                    mGraphic.currentCamera.moveEye(0, moveDistance, 0, MyCamera3D.IN_CAMERA_SPACE);
                    break;
                case CubeGroup.AROUND_Z:
                    mGraphic.currentCamera.moveEye(0, 0, moveDistance, MyCamera3D.IN_CAMERA_SPACE);
                    break;
            }
        }
    }
}
