package me.ylab.rc.scenes {
    import flash.text.TextField;
    
    import me.ylab.application.IScene;
    import me.ylab.application.MyLogic;
    import me.ylab.application.MyScene;
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.tools.IMyKeyEventListener;
    import me.ylab.tools.IMyMouseEventListener;

    public class CubeBaseScene extends MyLogic implements IScene, IMyMouseEventListener, IMyKeyEventListener {

        protected var mEventListeners:Vector.<Vector.<Function>>;
        protected var mGraphic:Graphic3D;
        protected var mFPSLabel:TextField;

        public function CubeBaseScene() {}
        
        public function get fpsLagel():TextField {
            return mFPSLabel;
        }

        public override function runLogic(lapseTime:int):void {
            mFPSLabel.text = 'FPS: ' + mGraphic.fps.toFixed(3);
            mGraphic.graphicLogic();
        }
        
        public function onKeyDown(key:uint):void {}
        
        public function onKeyUp(key:uint):void {}
        
        public function onMouseDown(x:int, y:int):void {}
        
        public function onMouseMove(x:int, y:int):void {}
        
        public function onMouseUp(x:int, y:int):void {}
        
        public function onMouseWheel(d:int):void {}
        
        public function addEventListener(event:int, fun:Function):void {
            if (event < MyScene.ON_LOAD_COMPLETE || event > MyScene.ON_SHIFT_COMPLETE) return;
            mEventListeners[event].push(fun);
        }
        
        public function getKeyListener():IMyKeyEventListener {
            return this;
        }
        
        public function getLogic():MyLogic {
            return this;
        }
        
        public function getMouseListener():IMyMouseEventListener {
            return this;
        }
        
        public function getShiftToScene():IScene {
            return null;
        }
        
        public function load():void {}
        
        public function prepare():void {}
        
        public function removeEventListener(event:int, fun:Function):void {
            if (event < MyScene.ON_LOAD_COMPLETE || event > MyScene.ON_SHIFT_COMPLETE) return;
            var pos:int = mEventListeners[event].indexOf(fun);
            if (pos < 0) return;
            mEventListeners[event].splice(pos, 1);
        }
        
        protected function notifyEvent(event:int):void {
            for each (var f:Function in mEventListeners[event]) {
                f.call(null, this);
            }
        }
        
        public function shiftTo(scene:IScene):void {}
        
        public function unload():void {}
        
        public function unprepare():void {}
        
    }
}
