package me.ylab.rc.scenes {
    
    import flash.display.Stage;
    import flash.external.ExternalInterface;
    import flash.text.TextField;
    
    import me.ylab.application.IScene;
    import me.ylab.application.MyAnimation;
    import me.ylab.application.MyLogic;
    import me.ylab.application.MyScene;
    import me.ylab.graphic.component.MyTextTexture;
    import me.ylab.graphic.foundation.CrossTestResult;
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.foundation.MyCamera3D;
    import me.ylab.graphic.foundation.MyColor4f;
    import me.ylab.graphic.foundation.MyModel;
    import me.ylab.graphic.types.TexturedVertex3;
    import me.ylab.graphic.types.indices.textured.MyTexturedVertexIndicesMesh;
    import me.ylab.graphic.types.indices.textured.MyTexturedVertexIndicesTriangle;
    import me.ylab.rc.CubeApplication;
    import me.ylab.rc.CubeGroup;
    import me.ylab.rc.animations.CubeOperateAnimation;
    import me.ylab.tools.IMyKeyEventListener;
    import me.ylab.tools.IMyMouseEventListener;
    import me.ylab.tools.MyKeyEventCode;
    import me.ylab.tools.MyVertex2;
    
    public class CubeHelpMenuScene extends CubeBaseScene {
        
        private var mStage:Stage;
        private var mApp:CubeApplication;
        private var mCubeGroup:CubeGroup;
        
        private var mMainMenuScene:CubeMainMenuScene;
        private var mShiftToMainMenuAnimation:MyAnimation;
        private var mShiftToScene:IScene;

        private var mHelpTextTexture:MyTextTexture;
        private var mHelpTextModel:MyModel;
        private var mHelpBaseModel:MyModel;
        private var mReturnTexture:MyTextTexture;
        private var mReturnModel:MyModel;
        
        protected var mIsMouseDown:Boolean;
        protected var mLastMousePos:MyVertex2;
        
        public function CubeHelpMenuScene(stage:Stage, graphic:Graphic3D, app:CubeApplication, cubeGroup:CubeGroup, mainMenuScene:CubeMainMenuScene, fpsLabel:TextField) {
            mEventListeners = new Vector.<Vector.<Function>>;
            mEventListeners[MyScene.ON_LOAD_COMPLETE] = new Vector.<Function>;
            mEventListeners[MyScene.ON_UNLOAD_COMPLETE] = new Vector.<Function>;
            mEventListeners[MyScene.ON_SHIFT_COMPLETE] = new Vector.<Function>;
            
            mStage = stage;
            mGraphic = graphic;
            mApp = app;
            mCubeGroup = cubeGroup;
            mMainMenuScene = mainMenuScene;
            mFPSLabel = fpsLabel;
            
            mHelpTextTexture = new MyTextTexture(mGraphic);
            mReturnTexture = new MyTextTexture(mGraphic);
            mHelpTextTexture.setText("帮助： \n\n    在方块上点击拖动，可以旋\n转一个面。\n    在其它地方点击拖动，旋转\n整个魔方。", MyColor4f.white, 20, "simhei");
            mReturnTexture.setText("返  回", MyColor4f.white, 20);
            
            // help text
            mHelpTextModel = new MyModel(mGraphic);
            mHelpTextModel.addMesh(new MyTexturedVertexIndicesMesh(mGraphic, null, null, mHelpTextTexture));
            mHelpTextModel.coordinate.insertTranslate(0, 0.7, 1.61);
            mHelpTextModel.coordinate.insertRotateByY(90);
            
            // return button
            mReturnModel = new MyModel(mGraphic);
            mReturnModel.addMesh(new MyTexturedVertexIndicesMesh(mGraphic, null, null, mReturnTexture));
            mReturnModel.coordinate.insertTranslate(0, -0.9, 1.61);
            
            mLastMousePos = new MyVertex2;
        }
        
        public override function load():void {
            mHelpBaseModel = mCubeGroup.cubes[14].model;

            if (mHelpBaseModel.subModels.indexOf(mHelpTextModel) < 0) {
                mGraphic.addModel(mHelpTextModel);
                mHelpTextModel.transformTo(mHelpBaseModel);
            }
            
            notifyEvent(MyScene.ON_LOAD_COMPLETE);
        }
        
        public override function unload():void {
            mHelpTextModel.transformTo(mGraphic.world);
            mHelpTextModel.father.removeModel(mHelpTextModel);
        }
        
        public override function prepare():void {
            mGraphic.addModel(mReturnModel);
        }
        
        public override function unprepare():void {
            mReturnModel.father.removeModel(mReturnModel);
        }
        
        public override function shiftTo(scene:IScene):void {
            switch (scene) {
            case mMainMenuScene:
                mShiftToMainMenuAnimation = new CubeOperateAnimation(mCubeGroup, null, CubeGroup.AROUND_Y, -90, 0, 5);
                mShiftToMainMenuAnimation.addEventListener(MyAnimation.ON_ANIMATION_END, onRotateFinish);
                mAnimationManager.add(mShiftToMainMenuAnimation);
                mShiftToMainMenuAnimation.start();
                mShiftToScene = mMainMenuScene;
                break;
            case mMainMenuScene.viewScene:
                mCubeGroup.helperModel.coordinate.insertRotateByY(90);
                mShiftToScene = mMainMenuScene.viewScene;
                unprepare();
                mMainMenuScene.prepare();
                mMainMenuScene.unload();
                mGraphic.currentCamera.moveEye(0, 0, 3, MyCamera3D.IN_CAMERA_SPACE);
                mMainMenuScene.viewScene.load();
                mMainMenuScene.viewScene.prepare();
                notifyEvent(MyScene.ON_SHIFT_COMPLETE);
                break;
            case mMainMenuScene.solveScene:
                mCubeGroup.helperModel.coordinate.insertRotateByY(90);
                mShiftToScene = mMainMenuScene.solveScene;
                unprepare();
                mMainMenuScene.prepare();
                mMainMenuScene.unload();
                mGraphic.currentCamera.moveEye(0, 0, 3, MyCamera3D.IN_CAMERA_SPACE);
                mMainMenuScene.solveScene.load();
                mMainMenuScene.solveScene.prepare();
                notifyEvent(MyScene.ON_SHIFT_COMPLETE);
                ExternalInterface.call("onSolveStart");
                break;
            }

        }

        public override function getShiftToScene():IScene {
            return mShiftToScene;
        }

        private function onRotateFinish(ani:MyAnimation):void {
            ani.removeEventListener(MyAnimation.ON_ANIMATION_END, onRotateFinish);
            ani.stop();
            mAnimationManager.remove(ani);
            unprepare();
            mMainMenuScene.prepare();
            notifyEvent(MyScene.ON_SHIFT_COMPLETE);
        }
        
        public override function onMouseDown(x:int, y:int):void {
            mIsMouseDown = true;
            mLastMousePos.setXY(x, y);
            mGraphic.selectX = x;
            mGraphic.selectY = y;
            
            if (null != mShiftToMainMenuAnimation && mShiftToMainMenuAnimation.isAnimated) return;
            var tCrossRes:CrossTestResult = mGraphic.selectResult;
            if (tCrossRes.result) {
                if (tCrossRes.mesh.model == mReturnModel) {
                    mApp.sceneShiftTo(mMainMenuScene);
                }
            }
        }
        
        public override function onKeyUp(key:uint):void {
            switch (key) {
                case MyKeyEventCode.KEY_F:
                    if (mStage.contains(mFPSLabel)) {
                        mStage.removeChild(mFPSLabel);
                    } else {
                        mStage.addChild(mFPSLabel);
                    }
                    break;
            }
        }
    }
}
