package me.ylab.rc.scenes {
    import flash.display.SimpleButton;
    import flash.display.Stage;
    import flash.events.Event;
    import flash.events.MouseEvent;
    import flash.external.ExternalInterface;
    import flash.text.TextField;
    import flash.text.TextFieldAutoSize;
    
    import me.ylab.application.IScene;
    import me.ylab.application.MyAnimation;
    import me.ylab.application.MyLogic;
    import me.ylab.application.MyScene;
    import me.ylab.application.MyTimer;
    import me.ylab.graphic.component.SkyBox;
    import me.ylab.graphic.foundation.CrossTestResult;
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.foundation.MyCamera3D;
    import me.ylab.rc.Cube;
    import me.ylab.rc.CubeApplication;
    import me.ylab.rc.CubeGroup;
    import me.ylab.rc.animations.CubeOperateAnimation;
    import me.ylab.tools.IMyKeyEventListener;
    import me.ylab.tools.IMyMouseEventListener;

    import me.ylab.tools.MyVector4;

    import me.ylab.tools.MyVertex3;

    public class CubeViewScene extends CubeBaseScene {

        private var mStage:Stage;
        private var mApp:CubeApplication;

        protected var mCubeGroup:CubeGroup;
        protected var mSkyBox:SkyBox;
        protected var mIsMouseDown:Boolean;
        protected var mLastMousePos:MyVertex2;
        protected var mStartRotateCubeMousePos:MyVertex2;
        protected var mEndRotateCubeMousePos:MyVertex2;
        protected var mStartRotateCubeCrossTestResult:CrossTestResult;

        protected var mRotateType:int;
        protected var mTestRay:Boolean;
        
        private var mRotateCube:Boolean;
        private var mRotateCubeSensitivity:int;
        private var mRotateCubeDir:int;
        private var mRotateBaseVector:MyVector2;
        private var mRotateCubes:Vector.<Cube>;
        private var mStartRotate:Boolean;
        private var mRotateFixAnimation:CubeOperateAnimation;
        private var mTotalRotateAngle:int;

        private var mPauseMenuScene:CubePauseMenuScene;
        
        private var mShiftToScene:IScene;
        
        private var mPauseButton:SimpleButton;
        
        public function CubeViewScene(stage:Stage, graphic:Graphic3D, app:CubeApplication, cubeGroup:CubeGroup,
                                      pauseMenuScene:CubePauseMenuScene, fpsLabel:TextField, skyBox:SkyBox) {
            mEventListeners = new Vector.<Vector.<Function>>;
            mEventListeners[MyScene.ON_LOAD_COMPLETE] = new Vector.<Function>;
            mEventListeners[MyScene.ON_UNLOAD_COMPLETE] = new Vector.<Function>;
            mEventListeners[MyScene.ON_SHIFT_COMPLETE] = new Vector.<Function>;
            
            mStage = stage;
            mGraphic = graphic;
            mCubeGroup = cubeGroup;
            mSkyBox = skyBox;
            mFPSLabel = fpsLabel;
            
            mApp = app;
            mGraphic.selectState = true;
            mLastMousePos = new MyVertex2;
            mStartRotateCubeMousePos = new MyVertex2;
            mEndRotateCubeMousePos = new MyVertex2;
            mStartRotateCubeCrossTestResult = new CrossTestResult;
            mRotateBaseVector = new MyVector2;
            mRotateCubes = new Vector.<Cube>;
            mStartRotate = false;
            mRotateFixAnimation = null;
            
            mIsMouseDown = false;
            mRotateCube = false;
            mTestRay = false;
            mRotateType = MyCamera3D.FIXED_EYE_ON;
            
            mRotateCubeSensitivity = 3;
            
            mPauseButton = new SimpleButton();
            mPauseButton.x = mGraphic.currentCamera.viewPortSize.width - 30;
            mPauseButton.y = 0;
            var pauseText:TextField = new TextField;
            pauseText.text = "暂停";
            pauseText.width = 30;
            pauseText.height = 20;
            pauseText.textColor = 0xFFFFFF;
            mPauseButton.upState = pauseText;
            mPauseButton.downState = pauseText;
            mPauseButton.overState = pauseText;
            mPauseButton.hitTestState = pauseText;
            mPauseButton.addEventListener(MouseEvent.CLICK, onPauseButtonClick);
            
            mPauseMenuScene = pauseMenuScene;
            
        }

        public function set cubeGroup(group:CubeGroup):void {
            mCubeGroup = group;
        }
        
        public override function load():void {
            mStage.addChild(mPauseButton);
            mPauseMenuScene.viewScene = this;
            mCubeGroup.operateToBegin();
            mCubeGroup.setViewMode(true);
            ExternalInterface.call("onViewMode");
        }
        
        public override function unload():void {
            mCubeGroup.setViewMode(false);
            mStage.removeChild(mPauseButton);
        }
        
        public override function shiftTo(scene:IScene):void {
            switch (scene) {
            case mPauseMenuScene:
                mShiftToScene = mPauseMenuScene;
                unload();
                mPauseMenuScene.load();
                notifyEvent(MyScene.ON_SHIFT_COMPLETE);
                break;
            }
        }
        
        public override function getShiftToScene():IScene {
            return mShiftToScene;
        }
        
        private function onPauseButtonClick(e:Event):void {
            mApp.sceneShiftTo(mPauseMenuScene);
        }
        
        public override function onMouseDown(x:int, y:int):void {
            mIsMouseDown = true;
            mLastMousePos.setXY(x, y);
            mGraphic.selectX = x;
            mGraphic.selectY = y;
            mRotateCube = false;
        }
        
        public override function onMouseUp(x:int, y:int):void {
            mIsMouseDown = false;
            mStartRotate = false;
        }
        
        public override function onMouseMove(x:int, y:int):void {
            if (mIsMouseDown) {
                if (!mRotateCube) {
                    mGraphic.currentCamera.rotate(-(y - mLastMousePos.y) / 5, -(x - mLastMousePos.x) / 5, 0, mRotateType);
//                    mGraphic.currentCamera.rotateModelToFace(mSkyBox);
//                    mSkyBox.resetTranslate();
                }
            }
            mLastMousePos.setXY(x, y);
        }
        
        public override function onMouseWheel(d:int):void {
            if (d > 0 && mGraphic.currentCamera.eyeOn.z > -5) return;
            if (d < 0 && mGraphic.currentCamera.eyeOn.z < -15) return;
            mGraphic.currentCamera.moveEye(0, 0, -d / 10, MyCamera3D.IN_CAMERA_SPACE);
        }

        public override function onKeyUp(key:uint):void {
            switch (key) {
                case MyKeyEventCode.KEY_F:
                    if (mStage.contains(mFPSLabel)) {
                        mStage.removeChild(mFPSLabel);
                    } else {
                        mStage.addChild(mFPSLabel);
                    }
                    break;
            }
        }
    }
}
