        private function onTimer(timer:MyTimer):void {
            mTimerLabel.text = ((timer.hour < 10)?"0":"") + timer.hour.toString() + ":"
                            + ((timer.minute < 10)?"0":"") + timer.minute.toString() + ":"
                            + ((timer.second < 10)?"0":"") + timer.second.toString();
            mTimerLabel.x = (mGraphic.currentCamera.viewPortSize.width - mTimerLabel.width) / 2;
        }
        
        private function onResetTimer(timer:MyTimer):void {
            mTimerLabel.text = "00:00:00";

        }

        public override function shiftTo(scene:IScene):void {
            switch (scene) {
            case mPauseMenuScene:
                mTimer.pause();
                mShiftToScene = mPauseMenuScene;
                unload();
                mPauseMenuScene.load();
                notifyEvent(MyScene.ON_SHIFT_COMPLETE);
                break;
            case mFinishMenuScene:
                mTimer.pause();
                mShiftToScene = mFinishMenuScene;
                unload();
                mFinishMenuScene.load();
                notifyEvent(MyScene.ON_SHIFT_COMPLETE);
                ExternalInterface.call("onFinish", mTimerLabel.text);
                break;
            case viewScene:
                mShiftToScene = viewScene;
                mStage.removeChild(mTimerLabel);
                unprepare();
                unload();
                viewScene.load();
                viewScene.prepare();
                notifyEvent(MyScene.ON_SHIFT_COMPLETE);
                break;
            }
        }

        public override function onMouseDown(x:int, y:int):void {
            mIsMouseDown = true;
            mLastMousePos.setXY(x, y);
            mGraphic.selectX = x;
            mGraphic.selectY = y;
            if (null != mRotateFixAnimation) {
                mRotateCube = false;
                return;
            }
            var tCrossRes:CrossTestResult = mGraphic.selectResult;
            if (tCrossRes.result && null != mCubeGroup) {
                var crossCube:Cube = mCubeGroup.getCubeByModel(tCrossRes.mesh.model);
                if (null == crossCube) return;
                mRotateCube = true;
                mStartRotateCubeMousePos.setXY(x, -y);
                mStartRotateCubeCrossTestResult.resetBy(tCrossRes);
            } else {
                mRotateCube = false;
            }
        }
        
        public override function onMouseUp(x:int, y:int):void {
            mIsMouseDown = false;
            if (mStartRotate && null == mRotateFixAnimation) {
                var rotateAngle:Number = (x - mStartRotateCubeMousePos.x)*mRotateBaseVector.x
                    + (-y - mStartRotateCubeMousePos.y)*mRotateBaseVector.y;
                
                var endAngle:Number = 0;
                var minAngle:int = mCubeGroup.getRotateMinDegree(mRotateCubeDir);
                var halfMinAngle:int = minAngle / 2;
                if (rotateAngle > 0) {
                    endAngle = int((rotateAngle + halfMinAngle) / minAngle) * minAngle;
                } else if (rotateAngle < 0) {
                    endAngle = int((rotateAngle - halfMinAngle) / minAngle) * minAngle;
                }
                
                mTotalRotateAngle = endAngle;
                
                mRotateFixAnimation = new CubeOperateAnimation(mCubeGroup, mRotateCubes, mRotateCubeDir, rotateAngle, endAngle, (endAngle - rotateAngle) / 5);
                
                mRotateFixAnimation.addEventListener(MyAnimation.ON_ANIMATION_END, onFixAnimationEnd);
                
                mApp.logic.animationManager.add(mRotateFixAnimation);
                mRotateFixAnimation.start();
                
            }
            mStartRotate = false;
        }
        
        public override function onMouseMove(x:int, y:int):void {
            if (mIsMouseDown) {
                if (!mRotateCube || mCubeGroup.isViewMode()) {
                    mGraphic.currentCamera.rotate(-(y - mLastMousePos.y) / 5, -(x - mLastMousePos.x) / 5, 0, mRotateType);
//                    mGraphic.currentCamera.rotateModelToFace(mSkyBox);
//                    mSkyBox.resetTranslate();
                } else if (!mCubeGroup.isViewMode()) {
                    if (!mStartRotate) {
                        var crossCube:Cube = mCubeGroup.getCubeByModel(mStartRotateCubeCrossTestResult.mesh.model);
                        if (null != crossCube) {
                            var xOffset:int = Math.abs(x - mStartRotateCubeMousePos.x);
                            var yOffset:int = Math.abs(-y - mStartRotateCubeMousePos.y);
                            if ((xOffset + yOffset) > mRotateCubeSensitivity) {
                                mEndRotateCubeMousePos.setXY(x, -y);
                                mRotateCubeDir = getRotateDirection(mStartRotateCubeCrossTestResult, mStartRotateCubeMousePos, mEndRotateCubeMousePos);
                                
                                mRotateCubes.length = 0;
                                mCubeGroup.fillRotateCubes(mRotateCubes, crossCube, mRotateCubeDir);
                                
                                mCubeGroup.helperModel.coordinate.returnToOrigin();
                                for each (var cube:Cube in mRotateCubes) {
                                    cube.model.transformTo(mCubeGroup.helperModel);
                                }
                                mStartRotate = true;
                            }
                        }
                    } else {
                        var rotateAngle:Number = (x - mStartRotateCubeMousePos.x)*mRotateBaseVector.x
                            + (-y - mStartRotateCubeMousePos.y)*mRotateBaseVector.y;
                        
                        mCubeGroup.helperModel.coordinate.returnToOrigin();
                        switch (mRotateCubeDir) {
                            case CubeGroup.AROUND_X:
                                mCubeGroup.helperModel.coordinate.insertRotateByX(rotateAngle);
                                break;
                            case CubeGroup.AROUND_Y:
                                mCubeGroup.helperModel.coordinate.insertRotateByY(rotateAngle);
                                break;
                            case CubeGroup.AROUND_Z:
                                mCubeGroup.helperModel.coordinate.insertRotateByZ(rotateAngle);
                                break;
                        }
                    }
                }
            }
            mLastMousePos.setXY(x, y);
//            if (mTestRay) {
//                mGraphic.selectX = x;
//                mGraphic.selectY = y;
//                var tCrossRes:CrossTestResult = mGraphic.selectResult;
//            }
        }
        
        public override function onMouseWheel(d:int):void {
            if (d > 0 && mGraphic.currentCamera.eyeOn.z > -5) return;
            if (d < 0 && mGraphic.currentCamera.eyeOn.z < -15) return;
            mGraphic.currentCamera.moveEye(0, 0, -d / 10, MyCamera3D.IN_CAMERA_SPACE);
        }

        protected function onFixAnimationEnd(ani:MyAnimation):void {
            mRotateFixAnimation.stop();
            mApp.logic.animationManager.remove(mRotateFixAnimation);
            mRotateFixAnimation = null;
            for each (var cube:Cube in mRotateCubes) {
                cube.model.transformTo(mCubeGroup.world);
            }
            mCubeGroup.rotateCube(mCubeGroup.getCubeByModel(mStartRotateCubeCrossTestResult.mesh.model), mTotalRotateAngle, mRotateCubeDir);
            if (mCubeGroup.finished) {
//                mApp.sceneShiftTo(mFinishMenuScene);
                ExternalInterface.call("onFinish", mTimerLabel.text);
            }
        }

        private function getRotateDirection(crossResult:CrossTestResult, startPos:MyVertex2, endPos:MyVertex2):int {
            var rotateDir:int = 0;
            var screenCrossVertex:MyVector2 = new MyVector2;
            var screenCrossYUp:MyVector2 = new MyVector2;
            var screenCrossZRight:MyVector2 = new MyVector2;
            
            var crossTriangleCoordinate:MyCoordinateSystem3 = crossResult.mesh.obtainWorldCoordinate();
            var crossVertexInTriangle:MyVector4 = new MyVector4();
            crossVertexInTriangle.resetBy4(crossResult.vertex);
            
            mGraphic.currentCamera.worldToScreen(crossVertexInTriangle, screenCrossVertex);
            screenCrossVertex.y = -screenCrossVertex.y;
            
            crossTriangleCoordinate.worldToSelf.multiplyToVertex(crossVertexInTriangle);
            
            var crossVertexYUp:MyVector4 = new MyVector4();
            crossVertexYUp.resetBy4(crossVertexInTriangle);
            var crossVertexZRight:MyVector4 = new MyVector4();
            crossVertexZRight.resetBy4(crossVertexInTriangle);
            
            crossVertexYUp.y += 0.5;
            crossVertexZRight.z += 0.5;
            
            crossTriangleCoordinate.selfToWorld.multiplyToVertex(crossVertexYUp);
            crossTriangleCoordinate.selfToWorld.multiplyToVertex(crossVertexZRight);
            
            
            mGraphic.currentCamera.worldToScreen(crossVertexYUp, screenCrossYUp);
            mGraphic.currentCamera.worldToScreen(crossVertexZRight, screenCrossZRight);
            screenCrossYUp.y = -screenCrossYUp.y;
            screenCrossZRight.y = -screenCrossZRight.y;
            
            crossVertexYUp.setXYZ(crossVertexYUp.x - crossResult.vertex.x,
                crossVertexYUp.y - crossResult.vertex.y,
                crossVertexYUp.z - crossResult.vertex.z);
            
            crossVertexZRight.setXYZ(crossVertexZRight.x - crossResult.vertex.x,
                crossVertexZRight.y - crossResult.vertex.y,
                crossVertexZRight.z - crossResult.vertex.z);
            
            crossVertexYUp.normalize();
            crossVertexZRight.normalize();
            
            var yUpVector:MyVector2 = screenCrossYUp.minus(screenCrossVertex);
            var zRightVector:MyVector2 = screenCrossZRight.minus(screenCrossVertex);
            
            yUpVector.normalize();
            zRightVector.normalize();
            
            var moveVector:MyVector2 = new MyVector2(endPos.x - startPos.x, endPos.y - startPos.y);
            
            //            trace('moveVector ' + moveVector.toString());
            var yUpDotProduct:Number = yUpVector.dotProduct(moveVector);
            var maxDotProduct:Number = yUpDotProduct;
            mRotateBaseVector.setXY(yUpVector.x, yUpVector.y);
            rotateDir = CubeGroup.AROUND_Z;
            //            trace('yUpDotProduct ' + yUpVector + yUpDotProduct);
            
            var zRightDotProduct:Number = zRightVector.dotProduct(moveVector);
            if (zRightDotProduct > maxDotProduct) {
                maxDotProduct = zRightDotProduct;
                mRotateBaseVector.setXY(zRightVector.x, zRightVector.y);
                rotateDir = CubeGroup.AROUND_Y;
            }
            //            trace('zRightDotProduct ' + zRightVector + zRightDotProduct);
            
            yUpVector.setXY(-yUpVector.x, -yUpVector.y);
            zRightVector.setXY(-zRightVector.x, -zRightVector.y);
            
            var yDownDotProduct:Number = yUpVector.dotProduct(moveVector);
            if (yDownDotProduct > maxDotProduct) {
                maxDotProduct = yDownDotProduct;
                mRotateBaseVector.setXY(-yUpVector.x, -yUpVector.y);
                rotateDir = CubeGroup.AROUND_Z;
            }
            //            trace('yDownDotProduct ' + yUpVector + yDownDotProduct);
            
            var zLeftDotProduct:Number = zRightVector.dotProduct(moveVector);
            if (zLeftDotProduct > maxDotProduct) {
                maxDotProduct = zLeftDotProduct;
                mRotateBaseVector.setXY(-zRightVector.x, -zRightVector.y);
                rotateDir = CubeGroup.AROUND_Y;
            }
            //            trace('zLeftDotProduct ' + zRightVector + zLeftDotProduct);
            //            trace(mRotateBaseVector.toString());
            //            trace();
            
            if (CubeGroup.AROUND_Z == rotateDir) {
                mRotateBaseVector.setXY(-mRotateBaseVector.x, -mRotateBaseVector.y);
                if (crossVertexZRight.x > 0.5) {
                    return CubeGroup.AROUND_X;
                } else if (crossVertexZRight.x < -0.5) {
                    mRotateBaseVector.setXY(-mRotateBaseVector.x, -mRotateBaseVector.y);
                    return CubeGroup.AROUND_X;
                }
                if (crossVertexZRight.y > 0.5) {
                    return CubeGroup.AROUND_Y;
                } else if (crossVertexZRight.y < -0.5) {
                    mRotateBaseVector.setXY(-mRotateBaseVector.x, -mRotateBaseVector.y);
                    return CubeGroup.AROUND_Y;
                }
                if (crossVertexZRight.z > 0.5) {
                    return CubeGroup.AROUND_Z;
                } else if (crossVertexZRight.z < -0.5) {
                    mRotateBaseVector.setXY(-mRotateBaseVector.x, -mRotateBaseVector.y);
                    return CubeGroup.AROUND_Z;
                }
            } else {
                if (crossVertexYUp.x > 0.5) {
                    return CubeGroup.AROUND_X;
                } else if (crossVertexYUp.x < -0.5) {
                    mRotateBaseVector.setXY(-mRotateBaseVector.x, -mRotateBaseVector.y);
                    return CubeGroup.AROUND_X;
                }
                if (crossVertexYUp.y > 0.5) {
                    return CubeGroup.AROUND_Y;
                } else if (crossVertexYUp.y < -0.5) {
                    mRotateBaseVector.setXY(-mRotateBaseVector.x, -mRotateBaseVector.y);
                    return CubeGroup.AROUND_Y;
                }
                if (crossVertexYUp.z > 0.5) {
                    return CubeGroup.AROUND_Z;
                } else if (crossVertexYUp.z < -0.5) {
                    mRotateBaseVector.setXY(-mRotateBaseVector.x, -mRotateBaseVector.y);
                    return CubeGroup.AROUND_Z;
                }
            }
            
            return 0;
        }
