        private function onLoadComplete(scene:IScene):void {
            scene.removeEventListener(MyScene.ON_LOAD_COMPLETE, onLoadComplete);
            
            mScene = scene;
            this.logic = scene.getLogic();
            mCubeKeyCtrl = scene.getKeyListener();
            mCubeMouseCtrl = scene.getMouseListener();
            
            resume();
            mCubeGroup.prepareToDraw(mGraphic);
            mCubeGroup.setOrders(3, 3, 3);

            mStage.addEventListener(KeyboardEvent.KEY_DOWN, onKeyEvent);
            mStage.addEventListener(KeyboardEvent.KEY_UP, onKeyEvent);
            mStage.addEventListener(MouseEvent.MOUSE_DOWN, onMouseEvent);
            mStage.addEventListener(MouseEvent.MOUSE_UP, onMouseEvent);
            mStage.addEventListener(MouseEvent.MOUSE_MOVE, onMouseEvent);
            mStage.addEventListener(MouseEvent.MOUSE_WHEEL, onMouseEvent);
            
            mStage.addEventListener(Event.ENTER_FRAME, enterFrame);
        }
        
        public function sceneShiftTo(scene:IScene):void {
            mScene.addEventListener(MyScene.ON_SHIFT_COMPLETE, onSceneShiftComplete);
            mScene.shiftTo(scene);
        }
        
        private function onSceneShiftComplete(scene:IScene):void {
            mScene.removeEventListener(MyScene.ON_SHIFT_COMPLETE, onSceneShiftComplete);
            
            mScene = scene.getShiftToScene();
            this.logic = mScene.getLogic();
            mCubeKeyCtrl = mScene.getKeyListener();
            mCubeMouseCtrl = mScene.getMouseListener();
        }
        
        private function enterFrame(e:Event):void {
            run();
        }
        
        private function onKeyEvent(e:KeyboardEvent):void {
            switch (e.type) {
                case KeyboardEvent.KEY_DOWN:
                    mCubeKeyCtrl.onKeyDown(e.keyCode);
                    break;
                case KeyboardEvent.KEY_UP:
                    mCubeKeyCtrl.onKeyUp(e.keyCode);
                    break;
            }
        }
        
        private function onMouseEvent(e:MouseEvent):void {
            switch (e.type) {
                case MouseEvent.MOUSE_DOWN:
                    mCubeMouseCtrl.onMouseDown(e.stageX, e.stageY);
                    break;
                case MouseEvent.MOUSE_UP:
                    mCubeMouseCtrl.onMouseUp(e.stageX, e.stageY);
                    break;
                case MouseEvent.MOUSE_MOVE:
                    mCubeMouseCtrl.onMouseMove(e.stageX, e.stageY);
                    break;
                case MouseEvent.MOUSE_WHEEL:
                    mCubeMouseCtrl.onMouseWheel(e.delta);
                    break;
            }
        }
        
        private function randomRotateAnimation():void {
            // x
            var autoAnimation:CubeAutoRotateAnimation = null;
            var endAngle:int = (Math.random()>0.5)?360:-360;
            
            if (endAngle < 0) {
                autoAnimation = new CubeAutoRotateAnimation(mGraphic, 1, endAngle, -(Math.random() + 0.2));
            } else {
                autoAnimation = new CubeAutoRotateAnimation(mGraphic, 1, endAngle, (Math.random() + 0.2));
            }
            
            autoAnimation.loop = true;
            
            autoAnimation.addEventListener(MyAnimation.ON_ANIMATION_END, onRotateAnimationEnd);
            
            mLogic.animationManager.add(autoAnimation);
            autoAnimation.start();
            
            // y
            endAngle = (Math.random()>0.5)?360:-360;
            
            if (endAngle < 0) {
                autoAnimation = new CubeAutoRotateAnimation(mGraphic, 2, endAngle, -(Math.random() + 0.2));
            } else {
                autoAnimation = new CubeAutoRotateAnimation(mGraphic, 2, endAngle, (Math.random() + 0.2));
            }
            
            autoAnimation.loop = true;
            
            autoAnimation.addEventListener(MyAnimation.ON_ANIMATION_END, onRotateAnimationEnd);
            
            mLogic.animationManager.add(autoAnimation);
            autoAnimation.start();

            // z
            endAngle = (Math.random()>0.5)?360:-360;
            
            if (endAngle < 0) {
                autoAnimation = new CubeAutoRotateAnimation(mGraphic, 3, endAngle, -(Math.random() + 0.2));
            } else {
                autoAnimation = new CubeAutoRotateAnimation(mGraphic, 3, endAngle, (Math.random() + 0.2));
            }
            
            autoAnimation.loop = true;
            
            autoAnimation.addEventListener(MyAnimation.ON_ANIMATION_END, onRotateAnimationEnd);
            
            mLogic.animationManager.add(autoAnimation);
            autoAnimation.start();
        }
        
        private function onRotateAnimationEnd(ani:CubeAutoRotateAnimation):void {
            return;
            ani.stop();
            mLogic.animationManager.remove(ani);
        }
