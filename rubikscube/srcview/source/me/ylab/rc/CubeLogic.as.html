package me.ylab.rc {

    import flash.display.Stage;
    import flash.text.TextField;
    
    import me.ylab.application.MyLogic;
    import me.ylab.graphic.foundation.Graphic3D;
    import me.ylab.graphic.foundation.MyCamera3D;
    import me.ylab.tools.MyVertex2;
    
    public class CubeLogic extends MyLogic {

        private var mStage:Stage;
        private var mGraphic:Graphic3D;
        private var mLabel:TextField;
        
        public function CubeLogic(stage:Stage, graphic:Graphic3D) {
            super();
            mStage = stage;
            mGraphic = graphic;
            
            mLabel = new TextField();
            mLabel.textColor = 0xFFFFFF;
            mStage.addChild(mLabel);
        }
        
        public override function runLogic(lapseTime:int):void {
            mLabel.text = 'FPS: ' + mGraphic.fps.toString();
            
            mGraphic.graphicLogic();
            
        }
        
    }
}
